22/// Directory traversal using compio's dispatcher for iterative processing
23///
24/// This function implements iterative directory traversal using compio's dispatcher
25/// instead of recursion or manual worklists.
26#[allow(clippy::too_many_arguments)]
27#[allow(clippy::future_not_send)]
28#[allow(clippy::used_underscore_binding)]
29#[allow(clippy::missing_docs_in_private_items)]
30pub(super) async fn traverse_and_copy_directory_iterative(
31    initial_src: PathBuf,
32    initial_dst: PathBuf,
33    file_ops: &FileOperations,
34    _copy_method: CopyMethod,
35    stats: &mut DirectoryStats,
36    hardlink_tracker: &mut FilesystemTracker,
37    metadata_config: &MetadataConfig,
38    concurrency_config: &crate::cli::ConcurrencyConfig,
39    parallel_config: &crate::cli::ParallelCopyConfig,
40) -> Result<()> {
41    // Create a dispatcher for async operations
42    // Using Box::leak for &'static lifetime - dispatcher lives for program duration
43    // This is intentional: the dispatcher manages worker threads and should not be dropped
44    let dispatcher = Box::leak(Box::new(Dispatcher::new()?));
45
46    // Create Arc-wrapped FileOperations and configs for safe sharing across async tasks
47    // No more unsafe transmute needed!
48    let file_ops_arc = Arc::new(file_ops.clone());
49    let metadata_config_arc = Arc::new(metadata_config.clone());
50    let parallel_config_arc = Arc::new(parallel_config.clone());
51
52    // Wrap shared state in wrapper types for static lifetimes
53    let stats_value = std::mem::take(stats);
54    let shared_stats = Arc::new(SharedStats::new(&stats_value));
55    let shared_hardlink_tracker = Arc::new(std::mem::take(hardlink_tracker));
56
57    // Check FD limits and warn if too low
58    if let Ok(fd_limit) = check_fd_limits() {
59        if fd_limit < concurrency_config.max_files_in_flight as u64 {
60            warn!(
61                "FD limit ({}) is less than --max-files-in-flight ({}). Consider: ulimit -n {}",
62                fd_limit,
63                concurrency_config.max_files_in_flight,
64                concurrency_config.max_files_in_flight * 2
65            );
66        }
67    }
68
69    // Create adaptive concurrency controller from config options
70    // The controller owns its configuration and behavior (adapt vs fail)
71    let concurrency_options = concurrency_config.to_options();
72    let concurrency_controller = Arc::new(AdaptiveConcurrencyController::new(&concurrency_options));
73
74    // Process the directory
75    // Note: We clone Arc values here, but this is necessary because we need to
76    // unwrap them later to return the final stats. The clone increments ref count,
77    // but all child operations complete before we unwrap, so it's just +1/-1.
78    // Delegate to root wrapper which handles DirectoryFd setup
79    // Build traversal context
80    let ctx = TraversalContext {
81        file_ops: file_ops_arc,
82        copy_method: _copy_method,
83        stats: shared_stats.clone(),
84        hardlink_tracker: shared_hardlink_tracker.clone(),
85        concurrency_controller,
86        metadata_config: metadata_config_arc,
87        parallel_config: parallel_config_arc,
88        dispatcher,
89    };
90
91    let result = process_root_entry(initial_src, initial_dst, ctx).await;
92
93    // Restore the state
94    // This unwraps successfully because the function and all child operations have completed,
95    // so the only remaining reference is the one we kept above (from .clone())
96    *stats = Arc::try_unwrap(shared_stats)
97        .map_err(|_| {
98            SyncError::FileSystem(
99                "Failed to unwrap Arc<SharedStats> - multiple references exist".to_string(),
100            )
101        })?
102        .into_inner();
103    *hardlink_tracker = Arc::try_unwrap(shared_hardlink_tracker).map_err(|_| {
104        SyncError::FileSystem(
105            "Failed to unwrap Arc<FilesystemTracker> - multiple references exist".to_string(),
106        )
107    })?;
108
109    result
110}

112/// Process directory entry using compio's dispatcher for async operations
113///
114/// This is the core function that handles all types of directory entries (files, directories, symlinks)
115/// using compio's dispatcher for efficient async scheduling. It's designed to be called recursively
116/// through the dispatcher to handle nested directory structures.
117///
118/// # Architecture
119///
120/// 1. **Entry Type Detection**: Uses `ExtendedMetadata` to determine if entry is file/dir/symlink
121/// 2. **Directory Processing**: Creates destination directory and dispatches all child entries
122/// 3. **File Processing**: Handles hardlink detection and file copying
123/// 4. **Symlink Processing**: Copies symlinks with target preservation
124///
125/// # Key Features
126///
127/// - **Unified Entry Handling**: Single function handles all entry types
128/// - **Concurrent Child Processing**: All child entries processed concurrently via dispatcher
129/// - **Hardlink Detection**: Tracks inodes to detect and create hardlinks efficiently
130/// - **Error Propagation**: Errors are properly propagated up the call stack
131///
132/// # Parameters
133///
134/// * `dispatcher` - Static dispatcher for scheduling async operations
135/// * `src_path` - Source path of the directory entry
136/// * `dst_path` - Destination path for the entry
137/// * `file_ops` - File operations handler with `io_uring` support
138/// * `copy_method` - Copy method (e.g., `io_uring`, fallback)
139/// * `stats` - Shared statistics tracking (wrapped in Arc<Mutex<>>)
140/// * `hardlink_tracker` - Shared hardlink detection (wrapped in Arc<Mutex<>>)
141///
142/// # Returns
143///
144/// Returns `Ok(())` if the entry is processed successfully, or `Err(SyncError)` if processing fails.
145///
146/// # Errors
147///
148/// This function will return an error if:
149/// - Metadata retrieval fails
150/// - Directory creation fails
151/// - File copying fails
152/// - Symlink copying fails
153/// - Hardlink operations fail
154///
155/// Helper: Open parent directory as `DirectoryFd`
156///
157/// Extracts the common pattern of opening a path's parent as `DirectoryFd`.
158#[allow(clippy::future_not_send)]
159pub(super) async fn open_parent_dirfd(path: &Path) -> Result<Arc<compio_fs_extended::DirectoryFd>> {
160    let parent = path.parent().unwrap_or_else(|| Path::new("."));
161    let dir_fd = compio_fs_extended::DirectoryFd::open(parent)
162        .await
163        .map_err(|e| {
164            SyncError::FileSystem(format!(
165                "Failed to open parent directory {}: {e}",
166                parent.display()
167            ))
168        })?;
169    Ok(Arc::new(dir_fd))
170}

172/// Process root entry (wrapper that sets up `DirectoryFd` for TOCTOU-safe operations)
173#[allow(clippy::future_not_send)]
174pub(super) async fn process_root_entry(
175    src_path: PathBuf,
176    dst_path: PathBuf,
177    ctx: TraversalContext,
178) -> Result<()> {
179    let src_parent_dir = open_parent_dirfd(&src_path).await?;
180    let src_filename = src_path
181        .file_name()
182        .ok_or_else(|| SyncError::FileSystem("No filename".to_string()))?
183        .to_os_string();
184
185    let dst_parent_dir = open_parent_dirfd(&dst_path).await?;
186    let dst_filename = dst_path
187        .file_name()
188        .ok_or_else(|| SyncError::FileSystem("No filename".to_string()))?
189        .to_os_string();
190
191    let src = FileLocation {
192        path: src_path,
193        parent_dir: src_parent_dir,
194        filename: src_filename,
195    };
196
197    let dst = FileLocation {
198        path: dst_path,
199        parent_dir: dst_parent_dir,
200        filename: dst_filename,
201    };
202
203    process_directory_entry_with_compio(src, dst, ctx).await
204}

206/// Internal: Process directory entry recursively with TOCTOU-safe `DirectoryFd` operations
207///
208/// This is the main recursive function for directory traversal. It processes files, directories,
209/// and symlinks using compio's dispatcher pattern to avoid stack overflow and enable
210/// concurrent processing.
211///
212/// **REQUIRES `DirectoryFd`**: All `DirectoryFd` parameters are required (no Options).
213/// This enforces TOCTOU-safe operations at compile time. Use `process_root_entry` for root.
214///
215/// # Parameters
216///
217/// * `dispatcher` - Static dispatcher for concurrent operations
218/// * `src_path` - Source path (for error messages only)
219/// * `dst_path` - Destination path (for error messages only)
220/// * `file_ops` - File operations handler
221/// * `_copy_method` - Copy method to use
222/// * `stats` - Shared statistics tracker
223/// * `hardlink_tracker` - Shared hardlink tracker for inode-based detection
224/// * `concurrency_controller` - Adaptive concurrency limiter
225/// * `metadata_config` - Metadata preservation configuration
226/// * `parallel_config` - Parallel copy configuration
227/// * `src_parent_dir` - Source parent `DirectoryFd` (REQUIRED for TOCTOU safety)
228/// * `src_filename` - Source basename (REQUIRED)
229/// * `dst_parent_dir` - Destination parent `DirectoryFd` (REQUIRED for TOCTOU safety)
230/// * `dst_filename` - Destination basename (REQUIRED)
231///
232/// # Returns
233///
234/// Returns `Ok(())` if the entry was processed successfully, or `Err(SyncError)` if failed.
235///
236/// # Errors
237///
238/// Returns an error if:
239/// - Directory creation fails
240/// - Metadata retrieval fails
241/// - File copy operations fail
242/// - Symlink operations fail
243/// - Hardlink operations fail
244#[allow(clippy::future_not_send)]
245#[allow(clippy::too_many_lines)]
246pub(super) async fn process_directory_entry_with_compio(
247    src: FileLocation,
248    dst: FileLocation,
249    ctx: TraversalContext,
250) -> Result<()> {
251    // Clone controller before acquiring permit to avoid borrow/move conflict
252    let controller = Arc::clone(&ctx.concurrency_controller);
253
254    // Acquire permit from adaptive concurrency controller
255    // This prevents unbounded queue growth and adapts to resource constraints (e.g., FD exhaustion)
256    // The permit is held for the entire operation (directory, file, or symlink)
257    let _permit = controller.acquire().await;
258
259    // Get comprehensive metadata using io_uring statx via DirectoryFd
260    // âœ… ALWAYS uses DirectoryFd - no fallback, no path-based operations!
261    let extended_metadata = src.parent_dir.statx_full(src.filename.as_ref()).await?;
262
263    if extended_metadata.is_dir() {
264        // ========================================================================
265        // DIRECTORY PROCESSING: Handle directory entries
266        // ========================================================================
267        debug!("Processing directory: {}", src.path.display());
268
269        // Try to create destination directory (TOCTOU-safe: no exists() check!)
270        match compio::fs::create_dir(&dst.path).await {
271            Ok(()) => {
272                ctx.stats.increment_directories_created();
273            }
274            Err(e) if e.kind() == std::io::ErrorKind::AlreadyExists => {
275                // Something exists - verify it's actually a directory
276                let existing_metadata = compio::fs::metadata(&dst.path).await.map_err(|e| {
277                    SyncError::FileSystem(format!(
278                        "Failed to check existing path {}: {}",
279                        dst.path.display(),
280                        e
281                    ))
282                })?;
283
284                if !existing_metadata.is_dir() {
285                    return Err(SyncError::FileSystem(format!(
286                        "Cannot create directory {}: path exists but is not a directory (is_file: {}, is_symlink: {})",
287                        dst.path.display(),
288                        existing_metadata.is_file(),
289                        existing_metadata.is_symlink()
290                    )));
291                }
292
293                debug!("Directory already exists: {}", dst.path.display());
294            }
295            Err(e) => {
296                return Err(SyncError::FileSystem(format!(
297                    "Failed to create directory {}: {}",
298                    dst.path.display(),
299                    e
300                )));
301            }
302        }
303
304        // Open the destination directory immediately (for metadata and children)
305        let dst_dir_fd = Arc::new(
306            compio_fs_extended::DirectoryFd::open(&dst.path)
307                .await
308                .map_err(|e| {
309                    SyncError::FileSystem(format!(
310                        "Failed to open destination directory {}: {}",
311                        dst.path.display(),
312                        e
313                    ))
314                })?,
315        );
316
317        // ALWAYS preserve directory metadata (whether just created or already existed)
318        // This ensures metadata is synchronized even on re-sync operations
319        preserve_directory_metadata_fd(
320            &src.path,
321            &dst.path,
322            &dst_dir_fd,
323            &extended_metadata,
324            &ctx.metadata_config,
325        )
326        .await?;
327
328        // Open source directory as DirectoryFd for TOCTOU-safe operations
329        let src_dir = Arc::new(
330            compio_fs_extended::DirectoryFd::open(&src.path)
331                .await
332                .map_err(|e| {
333                    SyncError::FileSystem(format!(
334                        "Failed to open source directory {}: {e}",
335                        src.path.display()
336                    ))
337                })?,
338        );
339
340        // Read directory entries using compio-fs-extended wrapper
341        // This abstracts whether read_dir is blocking or uses io_uring (currently blocking due to kernel limitation)
342        // See: compio_fs_extended::directory::read_dir for implementation details and kernel status
343        let entries = compio_fs_extended::directory::read_dir(&src.path)
344            .await
345            .map_err(|e| {
346                SyncError::FileSystem(format!(
347                    "Failed to read directory {}: {}",
348                    src.path.display(),
349                    e
350                ))
351            })?;
352
353        // ========================================================================
354        // CONCURRENT PROCESSING: Dispatch all child entries concurrently
355        // ========================================================================
356        // Collect all async operations to dispatch
357        let mut futures = Vec::new();
358
359        // Process each child entry using compio's dispatcher
360        // This is the key innovation: instead of recursion or manual worklists,
361        // we dispatch all child entries to the same function, creating a tree
362        // of concurrent operations that compio manages efficiently
363        let _copy_method = ctx.copy_method.clone();
364        for entry_result in entries {
365            let entry = entry_result.map_err(|e| {
366                SyncError::FileSystem(format!("Failed to read directory entry: {e}"))
367            })?;
368            let child_src_path = entry.path();
369            let file_name = child_src_path.file_name().ok_or_else(|| {
370                SyncError::FileSystem(format!("Invalid file name in {}", child_src_path.display()))
371            })?;
372            let child_dst_path = dst.path.join(file_name);
373            let file_name_osstring = file_name.to_os_string();
374
375            // Dispatch all entries to the same function regardless of type
376            // This creates a unified processing pipeline where each entry
377            // determines its own processing path (file/dir/symlink)
378            let child_src_path = child_src_path.clone();
379            let child_dst_path = child_dst_path.clone();
380            let ctx_clone = ctx.clone();
381            let src_dir_clone = Arc::clone(&src_dir);
382            let dst_dir_clone = Arc::clone(&dst_dir_fd);
383            let dst_file_name_osstring = file_name.to_os_string();
384
385            let child_src = FileLocation {
386                path: child_src_path.clone(),
387                parent_dir: src_dir_clone,
388                filename: file_name_osstring,
389            };
390
391            let child_dst = FileLocation {
392                path: child_dst_path.clone(),
393                parent_dir: dst_dir_clone,
394                filename: dst_file_name_osstring,
395            };
396
397            let receiver = ctx
398                .dispatcher
399                .dispatch(move || {
400                    process_directory_entry_with_compio(child_src, child_dst, ctx_clone)
401                })
402                .map_err(|e| {
403                    SyncError::FileSystem(format!("Failed to dispatch entry processing: {e:?}"))
404                })?;
405            futures.push(receiver);
406        }
407
408        // ========================================================================
409        // ERROR HANDLING: Short-circuit on first error
410        // ========================================================================
411        // Use try_join_all to short-circuit on first error
412        // This is crucial for performance: we don't wait for all operations
413        // to complete before checking for errors. As soon as any operation
414        // fails, we cancel the remaining operations and return the error.
415        let _ = futures::future::try_join_all(futures.into_iter().map(|receiver| async move {
416            let _ = receiver.await.map_err(|e| {
417                SyncError::FileSystem(format!(
418                    "Failed to receive result from dispatched operation: {e:?}"
419                ))
420            })?;
421            Ok::<(), SyncError>(())
422        }))
423        .await?;
424    } else if extended_metadata.is_file() {
425        // ========================================================================
426        // FILE PROCESSING: Handle regular files with hardlink detection
427        // ========================================================================
428        // Files are processed with hardlink detection to avoid copying
429        // the same content multiple times when hardlinks exist
430        process_file(src, dst, extended_metadata, ctx).await?;
431    } else if extended_metadata.is_symlink() {
432        // ========================================================================
433        // SYMLINK PROCESSING: Handle symbolic links
434        // ========================================================================
435        if ctx.metadata_config.should_preserve_links() {
436            // Copy symlink as symlink (preserve target)
437            process_symlink(src.path, dst.path, &ctx.metadata_config, ctx.stats.clone()).await?;
438        } else {
439            // Dereference symlink: recursively process the target
440            // This handles files, directories, and even chains of symlinks correctly
441            debug!(
442                "Dereferencing symlink (will copy target): {}",
443                src.path.display()
444            );
445
446            // Read symlink target
447            let target = std::fs::read_link(&src.path).map_err(|e| {
448                SyncError::FileSystem(format!(
449                    "Failed to read symlink {}: {}",
450                    src.path.display(),
451                    e
452                ))
453            })?;
454
455            // Resolve target path (handle relative symlinks)
456            let target_path = if target.is_absolute() {
457                target
458            } else {
459                src.path
460                    .parent()
461                    .ok_or_else(|| {
462                        SyncError::FileSystem(format!(
463                            "Symlink has no parent: {}",
464                            src.path.display()
465                        ))
466                    })?
467                    .join(target)
468            };
469
470            // Recursively process the target (handles files, dirs, and symlink chains)
471            // Use process_root_entry since target path could be anywhere (needs own DirectoryFd setup)
472            let receiver = ctx
473                .dispatcher
474                .dispatch(move || process_root_entry(target_path, dst.path, ctx))
475                .map_err(|e| {
476                    SyncError::FileSystem(format!(
477                        "Failed to dispatch symlink target processing: {e:?}"
478                    ))
479                })?;
480
481            receiver.await.map_err(|e| {
482                SyncError::FileSystem(format!("Symlink target processing failed: {e:?}"))
483            })??;
484        }
485    }
486
487    Ok(())
488}

493/// Process a regular file during directory traversal.
494///
495/// This handles hardlink detection via `FilesystemTracker`, creating a
496/// hardlink when possible or copying file contents otherwise. On successful
497/// copy/link creation, it updates shared statistics and tracker state.
498///
499/// # Parameters
500/// - `src_path`: Source file path to process
501/// - `dst_path`: Destination file path
502/// - `metadata`: Extended source metadata used for decisions (size, inode, links)
503/// - `_file_ops`: File operations handle (reserved for future metadata work)
504/// - `_copy_method`: Copy method placeholder (currently unified to read/write)
505/// - `stats`: Shared stats accumulator updated on success/error
506/// - `hardlink_tracker`: Shared tracker for inode-based hardlink handling
507///
508/// # Errors
509/// Returns an error if hardlink handling fails in an unrecoverable way or if
510/// filesystem operations cannot be performed.
511pub(super) async fn process_file(
512    src: FileLocation,
513    dst: FileLocation,
514    metadata: compio_fs_extended::FileMetadata,
515    ctx: TraversalContext,
516) -> Result<()> {
517    debug!(
518        "Processing file: {} (link_count: {})",
519        src.path.display(),
520        metadata.nlink
521    );
522
523    let device_id = metadata.dev;
524    let inode_number = metadata.ino;
525    let link_count = metadata.nlink;
526
527    // RACE-FREE HARDLINK PATTERN: Register and wait if linker
528    let is_copier = ctx
529        .hardlink_tracker
530        .register_file(&src.path, &dst.path, device_id, inode_number, link_count)
531        .await;
532
533    if is_copier {
534        // We're the copier - copy the file and signal completion
535        debug!(
536            "Copying file content (hardlink copier): {}",
537            src.path.display()
538        );
539
540        // Copy file with DirectoryFd (TOCTOU-safe, compile-time enforced)
541        // CRITICAL: Capture result but don't propagate yet - must signal linkers first!
542        let copy_result = copy_file_internal(
543            &src.path,
544            &dst.path,
545            &ctx.metadata_config,
546            &ctx.parallel_config,
547            ctx.dispatcher,
548            &metadata,
549            &src.parent_dir,
550            src.filename.as_ref(),
551            &dst.parent_dir,
552            dst.filename.as_ref(),
553        )
554        .await;
555
556        // Signal waiting linkers BEFORE propagating errors (prevents deadlock!)
557        // Linkers must wake up regardless of copy success/failure
558        ctx.hardlink_tracker.signal_copy_complete(inode_number);
559
560        // Now propagate the copy result (after linkers are notified)
561        copy_result?;
562
563        ctx.stats.increment_files_copied();
564        ctx.stats.increment_bytes_copied(metadata.size);
565        debug!("Copied file and signaled linkers: {}", dst.path.display());
566    } else if link_count > 1 {
567        // We're a linker - waiting is already done inside register_file()
568        // Get dst_path and create hardlink
569        let original_dst = ctx
570            .hardlink_tracker
571            .get_dst_path_for_inode(inode_number)
572            .ok_or_else(|| {
573                SyncError::FileSystem(format!(
574                    "BUG: dst_path not set for hardlink inode {inode_number}. \
575                     This should never happen after register_file returns for linker."
576                ))
577            })?;
578
579        debug!(
580            "Creating hardlink {} â†’ {} (inode: {})",
581            dst.path.display(),
582            original_dst.display(),
583            inode_number
584        );
585
586        // Create hardlink (will naturally fail if copier failed to create dst file)
587        handle_existing_hardlink(&dst.path, &original_dst, inode_number, &ctx.stats).await?;
588    } else {
589        // Regular file (link_count == 1) - copy normally
590        debug!(
591            "Copying file content (non-hardlink): {}",
592            src.path.display()
593        );
594
595        // Copy file with DirectoryFd (TOCTOU-safe, compile-time enforced)
596        copy_file_internal(
597            &src.path,
598            &dst.path,
599            &ctx.metadata_config,
600            &ctx.parallel_config,
601            ctx.dispatcher,
602            &metadata,
603            &src.parent_dir,
604            src.filename.as_ref(),
605            &dst.parent_dir,
606            dst.filename.as_ref(),
607        )
608        .await?;
609
610        ctx.stats.increment_files_copied();
611        ctx.stats.increment_bytes_copied(metadata.size);
612        debug!("Copied file: {}", dst.path.display());
613    }
614
615    Ok(())
616}

618/// Handle creation of a hardlink when the inode has already been copied
619///
620/// This helper is invoked when a file's inode has been seen previously (i.e.,
621/// the file is part of a hardlink set). Instead of copying file contents again,
622/// it creates a hardlink in the destination that points to the original copied
623/// path. It also ensures the destination's parent directory exists and updates
624/// shared statistics accordingly.
625///
626/// # Parameters
627///
628/// - `dst_path`: Destination path where the hardlink should be created
629/// - `src_path`: Source path (used for logging and error context)
630/// - `inode_number`: The inode identifier for the file being processed
631/// - `stats`: Shared statistics tracker used to record successes/errors
632/// - `hardlink_tracker`: Tracker used to look up the original path for this inode
633///
634/// # Returns
635///
636/// Returns `Ok(())` if the hardlink was created successfully (or if an expected
637/// recovery path was handled), otherwise returns `Err(SyncError)`.
638///
639/// # Errors
640///
641/// This function will return an error if:
642/// - The original path associated with `inode_number` cannot be determined
643/// - The destination parent directory cannot be created when needed
644/// - The hardlink creation via `std::fs::hard_link` fails unexpectedly
645///
646/// # Side Effects
647///
648/// - Increments the files-copied counter on successful hardlink creation
649/// - Increments the error counter on failures
650#[allow(clippy::future_not_send)]
651pub(super) async fn handle_existing_hardlink(
652    dst_path: &Path,
653    original_dst: &Path,
654    _inode_number: u64,
655    stats: &Arc<SharedStats>,
656) -> Result<()> {
657    // Create destination directory if needed
658    if let Some(parent) = dst_path.parent() {
659        if !parent.exists() {
660            compio::fs::create_dir_all(parent).await.map_err(|e| {
661                SyncError::FileSystem(format!(
662                    "Failed to create parent directory {}: {}",
663                    parent.display(),
664                    e
665                ))
666            })?;
667        }
668    }
669
670    // Create hardlink using compio-fs-extended for io_uring operations
671    compio_fs_extended::hardlink::create_hardlink_at_path(original_dst, dst_path)
672        .await
673        .map_err(|e| {
674            SyncError::FileSystem(format!(
675                "Failed to create hardlink from {} to {}: {}",
676                original_dst.display(),
677                dst_path.display(),
678                e
679            ))
680        })?;
681
682    stats.increment_files_copied();
683    debug!(
684        "Created hardlink: {} -> {}",
685        dst_path.display(),
686        original_dst.display()
687    );
688
689    Ok(())
690}
