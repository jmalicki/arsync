42/// Preserve directory extended attributes from source to destination
43///
44/// This function preserves all extended attributes from the source directory to the destination directory
45/// using file descriptor-based operations for maximum efficiency and security.
46///
47/// # Arguments
48///
49/// * `src_path` - Source directory path
50/// * `dst_path` - Destination directory path
51///
52/// # Returns
53///
54/// `Ok(())` if all extended attributes were preserved successfully
55///
56/// # Errors
57///
58/// This function will return an error if:
59/// - Extended attributes cannot be read from source
60/// - Extended attributes cannot be written to destination
61/// - Permission is denied for xattr operations
62#[allow(clippy::future_not_send)]
63pub async fn preserve_directory_xattr(src_path: &Path, dst_path: &Path) -> Result<()> {
64    use compio_fs_extended::{ExtendedFile, XattrOps};
65
66    // Open source and destination directories for xattr operations
67    let src_dir = compio::fs::File::open(src_path).await.map_err(|e| {
68        SyncError::FileSystem(format!("Failed to open source directory for xattr: {e}"))
69    })?;
70    let dst_dir = compio::fs::File::open(dst_path).await.map_err(|e| {
71        SyncError::FileSystem(format!(
72            "Failed to open destination directory for xattr: {e}"
73        ))
74    })?;
75
76    // Convert to ExtendedFile to access xattr operations
77    let extended_src = ExtendedFile::from_ref(&src_dir);
78    let extended_dst = ExtendedFile::from_ref(&dst_dir);
79
80    // Get all extended attribute names from source directory
81    let Ok(xattr_names) = extended_src.list_xattr().await else {
82        // If xattr is not supported or no xattrs exist, that's fine
83        return Ok(());
84    };
85
86    // Copy each extended attribute
87    for name in xattr_names {
88        match extended_src.get_xattr(&name).await {
89            Ok(value) => {
90                if let Err(e) = extended_dst.set_xattr(&name, &value).await {
91                    // Log warning but continue with other xattrs
92                    tracing::warn!(
93                        "Failed to preserve directory extended attribute '{}': {}",
94                        name,
95                        e
96                    );
97                }
98            }
99            Err(e) => {
100                tracing::warn!(
101                    "Failed to read directory extended attribute '{}': {}",
102                    name,
103                    e
104                );
105            }
106        }
107    }
108
109    Ok(())
110}

112/// Preserve directory metadata using pre-opened `DirectoryFd` (TOCTOU-safe, efficient)
113///
114/// This function accepts an already-open `DirectoryFd` and uses it for all metadata operations.
115/// This is more efficient (1 open vs 3) and TOCTOU-safe.
116///
117/// # Parameters
118/// - `src_path`: Source path (for error messages and xattr operations)
119/// - `dst_path`: Destination path (for error messages and xattr operations)  
120/// - `dst_dir_fd`: Pre-opened destination `DirectoryFd`
121/// - `extended_metadata`: Source metadata to copy
122/// - `metadata_config`: What metadata to preserve
123///
124/// # Errors
125///
126/// Returns error if metadata operations (permissions, ownership, timestamps) fail
127#[allow(clippy::similar_names)]
128#[allow(clippy::future_not_send)]
129pub async fn preserve_directory_metadata_fd(
130    src_path: &Path,
131    dst_path: &Path,
132    dst_dir_fd: &compio_fs_extended::DirectoryFd,
133    extended_metadata: &compio_fs_extended::FileMetadata,
134    metadata_config: &MetadataConfig,
135) -> Result<()> {
136    use compio_fs_extended::OwnershipOps;
137
138    // Get underlying File from DirectoryFd for metadata operations
139    let dst_file = dst_dir_fd.as_file();
140
141    // Preserve directory permissions if requested
142    if metadata_config.should_preserve_permissions() {
143        let mode = extended_metadata.permissions();
144        let compio_permissions = compio::fs::Permissions::from_mode(mode);
145
146        // Use FD-based set_permissions (TOCTOU-safe, no umask interference)
147        dst_file
148            .set_permissions(compio_permissions)
149            .await
150            .map_err(|e| {
151                SyncError::FileSystem(format!("Failed to preserve directory permissions: {e}"))
152            })?;
153
154        debug!(
155            "Preserved directory permissions for {}: {:o}",
156            dst_path.display(),
157            mode
158        );
159    }
160
161    // Preserve directory ownership if requested
162    if metadata_config.should_preserve_ownership() {
163        let source_uid = extended_metadata.uid;
164        let source_gid = extended_metadata.gid;
165
166        // Use FD-based fchown (TOCTOU-safe!)
167        dst_file.fchown(source_uid, source_gid).await.map_err(|e| {
168            SyncError::FileSystem(format!("Failed to preserve directory ownership: {e}"))
169        })?;
170
171        debug!(
172            "Preserved directory ownership for {}: uid={}, gid={}",
173            dst_path.display(),
174            source_uid,
175            source_gid
176        );
177    }
178
179    // Preserve directory timestamps if requested
180    if metadata_config.should_preserve_timestamps() {
181        let src_accessed = extended_metadata.accessed;
182        let src_modified = extended_metadata.modified;
183
184        // Use DirectoryFd::set_times (TOCTOU-safe!)
185        dst_dir_fd
186            .set_times(src_accessed, src_modified)
187            .await
188            .map_err(|e| {
189                SyncError::FileSystem(format!("Failed to preserve directory timestamps: {e}"))
190            })?;
191
192        debug!("Preserved directory timestamps for {}", dst_path.display());
193    }
194
195    // Preserve directory extended attributes if requested
196    if metadata_config.should_preserve_xattrs() {
197        preserve_directory_xattr(src_path, dst_path).await?;
198        debug!("Preserved directory xattrs for {}", dst_path.display());
199    }
200
201    Ok(())
202}

204/// Preserve directory metadata (legacy path-based wrapper)
205///
206/// **DEPRECATED**: Use `preserve_directory_metadata_fd` instead
207/// This wrapper exists for compatibility but opens the directory 3 times (inefficient!)
208///
209/// # Errors
210/// Returns error if `DirectoryFd` cannot be opened or metadata preservation fails
211#[allow(clippy::future_not_send)]
212pub async fn preserve_directory_metadata(
213    src_path: &Path,
214    dst_path: &Path,
215    extended_metadata: &compio_fs_extended::FileMetadata,
216    metadata_config: &MetadataConfig,
217) -> Result<()> {
218    // Open DirectoryFd once and use FD-based operations
219    let dst_dir_fd = compio_fs_extended::DirectoryFd::open(dst_path)
220        .await
221        .map_err(|e| SyncError::FileSystem(format!("Failed to open destination directory: {e}")))?;
222
223    preserve_directory_metadata_fd(
224        src_path,
225        dst_path,
226        &dst_dir_fd,
227        extended_metadata,
228        metadata_config,
229    )
230    .await
