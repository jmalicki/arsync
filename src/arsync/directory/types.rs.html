1//! Core types for directory traversal and copying
2//!
3//! This module contains the data structures used throughout directory operations:
4//! - `FileLocation`: Groups path, parent `DirectoryFd`, and filename
5//! - `TraversalContext`: Shared state passed through recursion
6//! - `DirectoryStats`: Statistics tracking
7

17/// Location information for a file/directory with `DirectoryFd` context
18///
19/// Groups the path, parent `DirectoryFd`, and filename together. This triplet appears
20/// everywhere (src and dst) and should be kept together for TOCTOU-safe operations.
21#[derive(Clone)]
22pub struct FileLocation {
23    /// Full path (used only for error messages and logging)
24    pub path: PathBuf,
25    /// Parent directory as `DirectoryFd` (for TOCTOU-safe operations)
26    pub parent_dir: Arc<compio_fs_extended::DirectoryFd>,
27    /// Filename relative to `parent_dir` (basename only, no path separators)
28    pub filename: std::ffi::OsString,
29}
30
31/// Context passed through directory traversal operations
32///
33/// Groups all the near-global state that needs to be passed down the recursion tree.
34/// This significantly reduces function argument counts and makes the code more maintainable.
35///
36/// # Benefits
37/// - Reduces function signatures from 15+ args to ~4-5 args
38/// - Easier to add new context in the future
39/// - Groups logically related state together
40/// - Makes function calls more readable
41#[derive(Clone)]
42pub struct TraversalContext {
43    /// File operations handler (contains copy method)
44    #[allow(dead_code)] // Reserved for future use
45    pub file_ops: Arc<FileOperations>,
46    /// Copy method to use (e.g., auto, `copy_file_range`, splice)
47    pub copy_method: CopyMethod,
48    /// Shared statistics accumulator
49    pub stats: Arc<crate::stats::SharedStats>,
50    /// Hardlink tracker for inode-based deduplication
51    pub hardlink_tracker: Arc<crate::hardlink_tracker::FilesystemTracker>,
52    /// Adaptive concurrency controller (prevents FD exhaustion)
53    pub concurrency_controller: Arc<AdaptiveConcurrencyController>,
54    /// Metadata preservation configuration
55    pub metadata_config: Arc<MetadataConfig>,
56    /// Parallel copy configuration
57    pub parallel_config: Arc<crate::cli::ParallelCopyConfig>,
58    /// Global dispatcher for parallel operations
59    pub dispatcher: &'static Dispatcher,
60}
61
62/// Directory copy operation statistics
63#[derive(Debug, Default)]
64pub struct DirectoryStats {
65    /// Total number of files copied
66    pub files_copied: u64,
67    /// Total number of directories created
68    pub directories_created: u64,
69    /// Total number of bytes copied
70    pub bytes_copied: u64,
71    /// Number of symlinks processed
72    pub symlinks_processed: u64,
73    /// Number of errors encountered
74    pub errors: u64,
75}

80/// Create FileMetadata from path (fallback - path-based, TOCTOU-vulnerable)
81///
82/// ⚠️ DEPRECATED: This uses path-based metadata which is TOCTOU-vulnerable.
83/// Prefer using `DirectoryFd::statx_full()` when possible.
84///
85/// # Errors
86///
87/// Returns error if metadata cannot be retrieved.
88#[allow(clippy::future_not_send)]
89pub async fn metadata_from_path(path: &Path) -> Result<compio_fs_extended::FileMetadata> {
90    use std::os::unix::fs::MetadataExt;
91
92    let compio_metadata = compio::fs::symlink_metadata(path).await.map_err(|e| {
93        SyncError::FileSystem(format!(
94            "Failed to get metadata for {}: {}",
95            path.display(),
96            e
97        ))
98    })?;
99
100    Ok(compio_fs_extended::FileMetadata {
101        size: compio_metadata.len(),
102        mode: compio_metadata.mode(),
103        uid: compio_metadata.uid(),
104        gid: compio_metadata.gid(),
105        nlink: compio_metadata.nlink(),
106        ino: compio_metadata.ino(),
107        dev: compio_metadata.dev(),
108        accessed: compio_metadata.accessed().unwrap_or(std::time::UNIX_EPOCH),
109        modified: compio_metadata.modified().unwrap_or(std::time::UNIX_EPOCH),
110        created: compio_metadata.created().ok(),
111        #[cfg(target_os = "linux")]
112        attributes: None,
113        #[cfg(target_os = "linux")]
114        attributes_mask: None,
115        #[cfg(target_os = "macos")]
116        flags: None,
117        #[cfg(target_os = "macos")]
118        generation: None,
119    })
120}
