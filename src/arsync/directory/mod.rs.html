37/// Copy an entire directory tree from source to destination
38///
39/// Recursively copies all files, directories, and symlinks from `src` to `dst`,
40/// preserving metadata according to the configuration in `args`. Uses async
41/// `io_uring` operations for optimal performance and TOCTOU-safe DirectoryFd-based
42/// operations for security.
43///
44/// # Arguments
45///
46/// * `src` - Source directory path to copy from
47/// * `dst` - Destination directory path to copy to
48/// * `file_ops` - File operations handler containing copy configuration
49/// * `_copy_method` - Copy method (e.g., auto, `copy_file_range`, splice)
50/// * `args` - Command-line arguments containing metadata and concurrency config
51///
52/// # Returns
53///
54/// Returns `DirectoryStats` containing operation counts (files copied, directories
55/// created, bytes transferred, symlinks processed, hardlinks detected).
56///
57/// # Errors
58///
59/// Returns error if:
60/// - Source directory doesn't exist or isn't accessible
61/// - Destination directory can't be created
62/// - File/directory operations fail during traversal
63/// - Metadata preservation fails
64#[allow(clippy::future_not_send)]
65#[allow(clippy::used_underscore_binding)]
66pub async fn copy_directory(
67    src: &Path,
68    dst: &Path,
69    file_ops: &FileOperations,
70    _copy_method: CopyMethod,
71    args: &Args,
72) -> Result<DirectoryStats> {
73    let mut stats = DirectoryStats::default();
74    let mut hardlink_tracker = FilesystemTracker::new();
75
76    info!(
77        "Starting directory copy from {} to {}",
78        src.display(),
79        dst.display()
80    );
81
82    // Create destination directory if it doesn't exist
83    if dst.exists() {
84        // Set source filesystem from root directory (destination already exists)
85        let root_metadata = types::metadata_from_path(src).await?;
86        hardlink_tracker.set_source_filesystem(root_metadata.dev);
87    } else {
88        compio::fs::create_dir_all(dst).await.map_err(|e| {
89            SyncError::FileSystem(format!(
90                "Failed to create destination directory {}: {}",
91                dst.display(),
92                e
93            ))
94        })?;
95        stats.directories_created += 1;
96        debug!("Created destination directory: {}", dst.display());
97
98        // Preserve root directory metadata (permissions, ownership, timestamps) if requested
99        let root_metadata = types::metadata_from_path(src).await?;
100        metadata::preserve_directory_metadata(src, dst, &root_metadata, &args.metadata).await?;
101
102        // Set source filesystem from root directory
103        hardlink_tracker.set_source_filesystem(root_metadata.dev);
104    }
105
106    // Traverse source directory iteratively using compio's dispatcher
107    traversal::traverse_and_copy_directory_iterative(
108        src.to_path_buf(),
109        dst.to_path_buf(),
110        file_ops,
111        _copy_method,
112        &mut stats,
113        &mut hardlink_tracker,
114        &args.metadata,
115        &args.concurrency,
116        &args.io.parallel,
117    )
118    .await?;
119
120    // Log hardlink detection results
121    let hardlink_stats = hardlink_tracker.get_stats();
122    info!(
123        "Directory copy completed: {} files, {} directories, {} bytes, {} symlinks",
124        stats.files_copied, stats.directories_created, stats.bytes_copied, stats.symlinks_processed
125    );
126    if hardlink_stats.hardlink_groups > 0 {
127        info!(
128            "Hardlink detection: {} unique files, {} hardlink groups, {} total hardlinks",
129            hardlink_stats.total_files,
130            hardlink_stats.hardlink_groups,
131            hardlink_stats.total_hardlinks
132        );
133    }
134
135    Ok(stats)
136}

361        use tempfile::TempDir;
362
363        let temp_dir = TempDir::new().expect("Failed to create temp directory");

365        let nonexistent = temp_dir.path().join("nonexistent.txt");
366
367        let stats = Arc::new(SharedStats::new(&DirectoryStats::default()));
368
369        // Try to create hardlink to nonexistent file - should fail
370        let result =

372
373        assert!(
374            result.is_err(),
375            "Creating hardlink to nonexistent file should fail"
376        );
377
378        // With clean error propagation, stats are NOT incremented on error
379        // Error propagates via Result instead
380        let final_stats = Arc::try_unwrap(stats).unwrap().into_inner();
381        assert_eq!(
382            final_stats.errors, 0,
383            "Error counter should NOT be incremented (error propagates via Result)"
384        );
385        assert_eq!(
386            final_stats.files_copied, 0,
387            "files_copied should not be incremented on failure"
388        );
