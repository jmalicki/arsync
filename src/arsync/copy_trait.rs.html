1//! Trait-based file copying
2//!
3//! This module provides file copying using the AsyncFile trait abstraction.
4//! It demonstrates how the trait can be used for file operations while
5//! maintaining compatibility with the existing copy infrastructure.
6
7use crate::error::{Result, SyncError};
8use crate::file_wrapper::AsyncFileWrapper;
9use crate::traits::AsyncFile;
10use compio::fs::File;
11use std::path::Path;
12
13/// Default buffer size for trait-based copying
14#[allow(dead_code)] // Used in later PRs
15const COPY_BUFFER_SIZE: usize = 64 * 1024; // 64KB
16
17/// Copy a file using the AsyncFile trait
18///
19/// This is a simple demonstration of using the trait-based file abstraction
20/// for file copying. It uses the AsyncFile trait methods to read from source
21/// and write to destination.
22///
23/// # Parameters
24///
25/// * `src` - Source file path
26/// * `dst` - Destination file path
27///
28/// # Returns
29///
30/// Returns `Ok(bytes_copied)` on success, where `bytes_copied` is the total
31/// number of bytes copied from source to destination.
32///
33/// # Errors
34///
35/// Returns `Err(SyncError)` if:
36/// - Source file cannot be opened or read
37/// - Destination file cannot be created or written
38/// - I/O error occurs during copy
39///
40/// # Examples
41///
42/// ```rust,ignore
43/// use arsync::copy_trait::copy_file_with_trait;
44/// use std::path::Path;
45///
46/// #[compio::main]
47/// async fn main() -> arsync::Result<()> {
48///     let bytes = copy_file_with_trait(
49///         Path::new("source.txt"),
50///         Path::new("dest.txt")
51///     ).await?;
52///     println!("Copied {} bytes", bytes);
53///     Ok(())
54/// }
55/// ```
56#[allow(dead_code)] // Used in later PRs
57pub async fn copy_file_with_trait(src: &Path, dst: &Path) -> Result<u64> {
58    // Open source file
59    let src_file = File::open(src)
60        .await
61        .map_err(|e| SyncError::FileSystem(format!("Failed to open source: {e}")))?;
62
63    // Wrap in AsyncFile trait
64    let src_wrapper = AsyncFileWrapper::new(src_file);
65
66    // Open destination file
67    let dst_file = File::create(dst)
68        .await
69        .map_err(|e| SyncError::FileSystem(format!("Failed to create destination: {e}")))?;
70
71    // Wrap in AsyncFile trait
72    let mut dst_wrapper = AsyncFileWrapper::new(dst_file);
73
74    // Get source file size
75    let total_size = src_wrapper.size().await?;
76
77    // Copy in chunks
78    let mut bytes_copied: u64 = 0;
79    let mut buffer = vec![0u8; COPY_BUFFER_SIZE];
80
81    while bytes_copied < total_size {
82        // Read chunk from source
83        let (bytes_read, returned_buffer) = src_wrapper.read_at(buffer, bytes_copied).await?;
84
85        if bytes_read == 0 {
86            break; // EOF
87        }
88
89        // Write chunk to destination
90        let chunk_to_write = returned_buffer[..bytes_read].to_vec();
91        let (bytes_written, returned_buffer) =
92            dst_wrapper.write_at(chunk_to_write, bytes_copied).await?;
93
94        if bytes_written == 0 {
95            return Err(SyncError::FileSystem(
96                "Failed to write: no bytes written".to_string(),
97            ));
98        }
99
100        if bytes_written != bytes_read {
101            return Err(SyncError::FileSystem(format!(
102                "Incomplete write: wrote {} of {} bytes",
103                bytes_written, bytes_read
104            )));
105        }
106
107        bytes_copied += bytes_written as u64;
108        buffer = returned_buffer;
109    }
110
111    // Sync to disk
112    dst_wrapper.sync_all().await?;
113
114    Ok(bytes_copied)
115}

117#[cfg(test)]
118mod tests {
119    use super::*;
120    use std::io::Write;
121    use tempfile::NamedTempFile;
122
123    #[compio::test]
124    async fn test_copy_small_file() -> anyhow::Result<()> {
125        // Create source file
126        let mut src_file = NamedTempFile::new()?;
127        src_file.write_all(b"Hello, World!")?;
128        src_file.flush()?;
129
130        // Create destination
131        let dst_file = NamedTempFile::new()?;
132        let dst_path = dst_file.path().to_path_buf();
133        drop(dst_file); // Close it so we can write
134
135        // Copy using trait
136        let bytes = copy_file_with_trait(src_file.path(), &dst_path).await?;
137
138        assert_eq!(bytes, 13);
139
140        // Verify contents
141        let contents = std::fs::read(&dst_path)?;
142        assert_eq!(contents, b"Hello, World!");
143
144        Ok(())
145    }

171        // Create 1MB file
172        let mut src_file = NamedTempFile::new()?;
173        let data = vec![0x42u8; 1024 * 1024]; // 1MB of 'B's
174        src_file.write_all(&data)?;
175        src_file.flush()?;
176
177        // Create destination
178        let dst_file = NamedTempFile::new()?;
179        let dst_path = dst_file.path().to_path_buf();
180        drop(dst_file);
181
182        // Copy using trait
183        let bytes = copy_file_with_trait(src_file.path(), &dst_path).await?;
184
185        assert_eq!(bytes, 1024 * 1024);
186
187        // Verify contents
188        let contents = std::fs::read(&dst_path)?;
189        assert_eq!(contents.len(), 1024 * 1024);
190        assert!(contents.iter().all(|&b| b == 0x42));
191
192        Ok(())
193    }

195    #[compio::test]
196    async fn test_copy_matches_original() -> anyhow::Result<()> {
197        // Create source file with specific pattern
198        let mut src_file = NamedTempFile::new()?;
199        let pattern: Vec<u8> = (0..=255).cycle().take(10000).collect();
200        src_file.write_all(&pattern)?;
201        src_file.flush()?;
202
203        // Create destination
204        let dst_file = NamedTempFile::new()?;
205        let dst_path = dst_file.path().to_path_buf();
206        drop(dst_file);
207
208        // Copy using trait
209        let bytes = copy_file_with_trait(src_file.path(), &dst_path).await?;
210
211        assert_eq!(bytes, 10000);
212
213        // Verify exact match
214        let src_contents = std::fs::read(src_file.path())?;
215        let dst_contents = std::fs::read(&dst_path)?;
216        assert_eq!(src_contents, dst_contents);
217
218        Ok(())
