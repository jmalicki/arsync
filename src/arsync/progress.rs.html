1//! Progress tracking and reporting
2//!
3//! This module provides comprehensive progress tracking and reporting functionality
4//! for file synchronization operations. It offers real-time progress visualization,
5//! performance statistics, and detailed operation metrics.
6//!
7//! # Features
8//!
9//! - Real-time progress bars with customizable display formats
10//! - Comprehensive statistics tracking (files, bytes, time)
11//! - Support for both individual operations and batch tracking
12//! - Configurable progress bar styling and templates
13//! - Thread-safe progress updates and statistics
14//!
15//! # Usage
16//!
17//! ```rust,ignore
18//! use arsync::progress::ProgressTracker;
19//! use arsync::io_uring::CopyOperation;
20//! use std::path::PathBuf;
21//!
22//! let mut tracker = ProgressTracker::new();
23//! tracker.set_total(1024 * 1024); // 1MB total
24//!
25//! // During operation
26//! tracker.update(512 * 1024); // 512KB copied
27//!
28//! // Get statistics
29//! let stats = tracker.stats();
30//! println!("Progress: {} files, {} bytes", stats.files_copied, stats.bytes_copied);
31//!
32//! tracker.finish();
33//! ```
34//!
35//! # Performance Considerations
36//!
37//! - Progress updates are lightweight and don't impact copy performance
38//! - Statistics are updated atomically for thread safety
39//! - Progress bar rendering is optimized for minimal overhead
40//! - Memory usage is constant regardless of operation size
41//!
42//! # Thread Safety
43//!
44//! All progress tracking operations are thread-safe and can be used concurrently
45//! across multiple threads. Statistics are updated atomically to prevent race conditions.
46

52/// Progress tracker for file synchronization operations
53///
54/// This structure provides real-time progress tracking and reporting for file
55/// copying operations. It maintains detailed statistics and displays progress
56/// using customizable progress bars.
57///
58/// # Fields
59///
60/// * `progress_bar` - Visual progress indicator using indicatif
61/// * `files_copied` - Running count of files successfully processed
62/// * `bytes_copied` - Running total of bytes successfully copied
63///
64/// # Examples
65///
66/// Basic usage:
67/// ```rust
68/// use arsync::progress::ProgressTracker;
69///
70/// let mut tracker = ProgressTracker::new();
71/// tracker.set_total(1024 * 1024); // Set 1MB target
72/// tracker.update(512 * 1024);     // Update with 512KB progress
73/// tracker.finish();               // Complete the operation
74/// ```
75///
76/// Tracking individual operations:
77/// ```rust
78/// use arsync::progress::ProgressTracker;
79/// use arsync::io_uring::CopyOperation;
80/// use std::path::PathBuf;
81///
82/// let mut tracker = ProgressTracker::new();
83/// let src_path = PathBuf::from("source.txt");
84/// let dst_path = PathBuf::from("destination.txt");
85/// let file_size = 1024;
86/// let operation = CopyOperation::new(src_path, dst_path, file_size);
87/// tracker.track_operation(&operation);
88/// ```
89///
90/// # Performance Notes
91///
92/// - Progress updates are O(1) operations
93/// - Statistics are maintained in atomic operations
94/// - Progress bar rendering is optimized for minimal CPU usage
95/// - Memory footprint is constant regardless of operation size
96#[derive(Debug)]
97#[allow(dead_code)]
98pub struct ProgressTracker {
99    /// Visual progress bar for user feedback
100    progress_bar: ProgressBar,
101
102    /// Total number of files discovered during traversal (via statx)
103    files_discovered: u64,
104    /// Total number of bytes discovered during traversal (via statx)
105    bytes_discovered: u64,
106
107    /// Running count of files successfully processed/copied
108    files_copied: u64,
109    /// Running total of bytes successfully copied
110    bytes_copied: u64,
111}
112
113impl Default for ProgressTracker {
114    fn default() -> Self {
115        Self::new()
116    }
117}

119#[allow(dead_code)]
120impl ProgressTracker {
121    /// Create a new progress tracker instance
122    ///
123    /// This function initializes a new progress tracker with default styling
124    /// and zero statistics. The progress bar is configured with a modern
125    /// template showing elapsed time, progress bar, bytes copied, and ETA.
126    ///
127    /// # Returns
128    ///
129    /// Returns a new `ProgressTracker` instance ready for use.
130    ///
131    /// # Examples
132    ///
133    /// ```rust
134    /// use arsync::progress::ProgressTracker;
135    ///
136    /// let tracker = ProgressTracker::new();
137    /// ```
138    ///
139    /// # Performance Notes
140    ///
141    /// - Initialization is O(1) and very fast
142    /// - No memory allocation beyond the struct itself
143    /// - Progress bar styling is applied immediately
144    ///
145    /// # Panics
146    ///
147    /// This function will panic if the progress bar template is invalid.
148    /// This should never happen with the hardcoded template string.
149    #[allow(dead_code)]
150    #[allow(clippy::unwrap_used)]
151    #[must_use]
152    pub fn new() -> Self {
153        let pb = ProgressBar::new(0);
154        pb.set_style(
155            ProgressStyle::default_bar()
156                .template("{spinner:.green} [{elapsed_precise}] [{wide_bar:.cyan/blue}] {bytes}/{total_bytes} ({eta})")
157                .unwrap()
158                .progress_chars("#>-"),
159        );
160
161        Self {
162            progress_bar: pb,
163            files_discovered: 0,
164            bytes_discovered: 0,
165            files_copied: 0,
166            bytes_copied: 0,
167        }
168    }
169
170    /// Track a file discovered during traversal (via statx)
171    ///
172    /// This should be called for each file encountered during directory traversal.
173    /// It updates the discovery counters but doesn't affect the progress bar
174    /// until the file is actually copied.
175    ///
176    /// # Parameters
177    ///
178    /// * `file_size` - Size of the discovered file in bytes
179    ///
180    /// # Examples
181    ///
182    /// ```rust
183    /// use arsync::progress::ProgressTracker;
184    ///
185    /// let mut tracker = ProgressTracker::new();
186    /// tracker.track_discovery(1024); // Discovered a 1KB file
187    /// ```
188    #[allow(dead_code)]
189    pub fn track_discovery(&mut self, file_size: u64) {
190        self.files_discovered += 1;
191        self.bytes_discovered += file_size;
192        // Update progress bar total to reflect discovered bytes
193        self.progress_bar.set_length(self.bytes_discovered);
194    }
195
196    /// Set the total number of bytes to be processed
197    ///
198    /// This function configures the progress bar with the total expected
199    /// byte count, enabling accurate progress percentage calculations and
200    /// ETA estimates.
201    ///
202    /// # Parameters
203    ///
204    /// * `total_bytes` - Total number of bytes to be processed in the operation
205    ///
206    /// # Examples
207    ///
208    /// ```rust
209    /// use arsync::progress::ProgressTracker;
210    ///
211    /// let mut tracker = ProgressTracker::new();
212    /// tracker.set_total(1024 * 1024); // Set 1MB target
213    /// ```
214    ///
215    /// # Performance Notes
216    ///
217    /// - This is an O(1) operation
218    /// - Progress bar updates immediately to reflect the new total
219    /// - Setting total to 0 disables percentage and ETA calculations
220    pub fn set_total(&self, total_bytes: u64) {
221        self.progress_bar.set_length(total_bytes);
222    }
223
224    /// Update progress with additional bytes copied
225    ///
226    /// This function updates the progress tracker with the number of bytes
227    /// that have been successfully copied. It increments both the file count
228    /// and byte count, and updates the visual progress bar.
229    ///
230    /// # Parameters
231    ///
232    /// * `bytes` - Number of bytes that were successfully copied
233    ///
234    /// # Examples
235    ///
236    /// ```rust
237    /// use arsync::progress::ProgressTracker;
238    ///
239    /// let mut tracker = ProgressTracker::new();
240    /// tracker.set_total(1024 * 1024);
241    /// tracker.update(512 * 1024); // 512KB copied
242    /// ```
243    ///
244    /// # Performance Notes
245    ///
246    /// - This is an O(1) operation
247    /// - Progress bar updates are batched for performance
248    /// - Statistics are updated atomically
249    /// - Visual updates are throttled to prevent excessive redraws
250    pub fn update(&mut self, bytes: u64) {
251        self.bytes_copied += bytes;
252        self.files_copied += 1;
253        self.progress_bar.inc(bytes);
254    }
255
256    /// Mark the progress tracking operation as complete
257    ///
258    /// This function finalizes the progress tracking operation, displaying
259    /// a completion message and stopping the progress bar animation.
260    ///
261    /// # Examples
262    ///
263    /// ```rust
264    /// use arsync::progress::ProgressTracker;
265    ///
266    /// let mut tracker = ProgressTracker::new();
267    /// // ... perform operations ...
268    /// tracker.finish(); // Show completion message
269    /// ```
270    ///
271    /// # Performance Notes
272    ///
273    /// - This is an O(1) operation
274    /// - Progress bar animation is stopped immediately
275    /// - Completion message is displayed once
276    /// - Statistics remain available after completion
277    pub fn finish(&self) {
278        let message = TranslationKey::ProgressComplete
279            .get()
280            .unwrap_or_else(|_| "Complete".to_string());
281        self.progress_bar.finish_with_message(message);
282    }
283
284    /// Track progress for a specific copy operation
285    ///
286    /// This function updates the progress tracker with statistics from a
287    /// completed or in-progress copy operation. It's designed to work with
288    /// the `CopyOperation` structure for detailed operation tracking.
289    ///
290    /// # Parameters
291    ///
292    /// * `operation` - The copy operation to track, containing file size,
293    ///   bytes copied, and other relevant statistics
294    ///
295    /// # Examples
296    ///
297    /// ```rust
298    /// use arsync::progress::ProgressTracker;
299    /// use arsync::io_uring::CopyOperation;
300    /// use std::path::PathBuf;
301    ///
302    /// let mut tracker = ProgressTracker::new();
303    /// let src_path = PathBuf::from("source.txt");
304    /// let dst_path = PathBuf::from("destination.txt");
305    /// let file_size = 1024;
306    /// let operation = CopyOperation::new(src_path, dst_path, file_size);
307    /// // ... perform copy operation ...
308    /// tracker.track_operation(&operation);
309    /// ```
310    ///
311    /// # Performance Notes
312    ///
313    /// - This is an O(1) operation
314    /// - Progress bar position is updated if file size is known
315    /// - Statistics are updated atomically
316    /// - Visual updates are optimized for minimal overhead
317    pub fn track_operation(&mut self, operation: &CopyOperation) {
318        self.files_copied += 1;
319        self.bytes_copied += operation.bytes_copied;
320
321        // Update progress bar if we have total size
322        if operation.file_size > 0 {
323            self.progress_bar.set_position(operation.bytes_copied);
324        }
325    }
326
327    /// Get current progress statistics
328    ///
329    /// This function returns a snapshot of the current progress statistics,
330    /// including file count, byte count, and elapsed time. The statistics
331    /// are captured at the time of the call and represent cumulative progress.
332    ///
333    /// # Returns
334    ///
335    /// Returns a `ProgressStats` structure containing:
336    /// - `files_copied`: Number of files successfully processed
337    /// - `bytes_copied`: Total bytes successfully copied
338    /// - `elapsed`: Time elapsed since tracker creation
339    ///
340    /// # Examples
341    ///
342    /// ```rust
343    /// use arsync::progress::ProgressTracker;
344    ///
345    /// let mut tracker = ProgressTracker::new();
346    /// // ... perform operations ...
347    /// let stats = tracker.stats();
348    /// println!("Progress: {} files, {} bytes, {:?} elapsed",
349    ///          stats.files_copied, stats.bytes_copied, stats.elapsed);
350    /// ```
351    ///
352    /// # Performance Notes
353    ///
354    /// - This is an O(1) operation
355    /// - Statistics are read atomically
356    /// - Elapsed time calculation is cached for performance
357    /// - No memory allocation occurs
358    #[must_use]
359    pub fn stats(&self) -> ProgressStats {
360        ProgressStats {
361            files_copied: self.files_copied,
362            bytes_copied: self.bytes_copied,
363            elapsed: self.progress_bar.elapsed(),
364        }
365    }
366}

368/// Progress statistics for synchronization operations
369///
370/// This structure contains comprehensive statistics about the progress
371/// of file synchronization operations, providing detailed metrics for
372/// monitoring, reporting, and performance analysis.
373///
374/// # Fields
375///
376/// * `files_copied` - Number of files successfully processed
377/// * `bytes_copied` - Total number of bytes successfully copied
378/// * `elapsed` - Time elapsed since operation started
379///
380/// # Examples
381///
382/// Basic usage:
383/// ```rust
384/// use arsync::progress::ProgressStats;
385/// use std::time::Duration;
386///
387/// let stats = ProgressStats {
388///     files_copied: 150,
389///     bytes_copied: 1_048_576,
390///     elapsed: Duration::from_secs(30),
391/// };
392///
393/// println!("Copied {} files ({} bytes) in {:?}",
394///          stats.files_copied,
395///          stats.bytes_copied,
396///          stats.elapsed);
397/// ```
398///
399/// Performance analysis:
400/// ```rust
401/// use arsync::progress::ProgressTracker;
402///
403/// let mut tracker = ProgressTracker::new();
404/// // ... perform operations ...
405/// let stats = tracker.stats();
406/// let throughput = stats.bytes_copied as f64 / stats.elapsed.as_secs_f64();
407/// println!("Throughput: {:.2} MB/s", throughput / 1_048_576.0);
408/// ```
409///
410/// # Performance Notes
411///
412/// - All fields are efficiently stored and accessed
413/// - Duration calculations are optimized for minimal overhead
414/// - Statistics can be safely shared across threads
415/// - Memory footprint is constant and minimal
416#[derive(Debug, Clone, PartialEq, Eq)]
417#[allow(dead_code)]
418pub struct ProgressStats {
419    /// Number of files successfully processed during the operation
420    pub files_copied: u64,
421
422    /// Total number of bytes successfully copied during the operation
423    pub bytes_copied: u64,
424
425    /// Time elapsed since the operation started
426    pub elapsed: Duration,
427}
