1//! rsync Protocol Handshake Implementation
2//!
3//! This module implements the rsync wire protocol handshake sequence, which establishes:
4//! 1. Protocol version compatibility between client and server
5//! 2. Capability negotiation (what features both sides support)
6//! 3. Random seed exchange for checksum algorithms
7//! 4. Multiplexing mode activation
8//!
9//! # Protocol Sequence
10//!
11//! ```text
12//! Client                                Server
13//!   |                                      |
14//!   |-- Version Byte (31) --------------->|
15//!   |<-------------- Version Byte (31) ---|
16//!   |                                      |
17//!   |-- Capability Flags (varint) ------->|
18//!   |<----- Capability Flags (varint) ----|
19//!   |                                      |
20//!   |-- Checksum Seed (optional) -------->|
21//!   |<----- Checksum Seed (optional) -----|
22//!   |                                      |
23//!   |   Handshake Complete                 |
24//!   |   (Multiplexing Active)              |
25//! ```
26//!
27//! # Example
28//!
29//! ```rust,no_run
30//! use arsync::protocol::handshake::{handshake, Role};
31//! use arsync::protocol::pipe::PipeTransport;
32//!
33//! # async fn example() -> anyhow::Result<()> {
34//! let mut transport = PipeTransport::from_stdio()?;
35//! let capabilities = handshake(&mut transport, Role::Sender).await?;
36//! println!("Negotiated protocol version: {}", capabilities.version);
37//! # Ok(())
38//! # }
39//! ```

117/// Role in the protocol handshake
118///
119/// The handshake is bidirectional, but each side has a different role:
120/// - **Sender**: Sends files to the receiver
121/// - **Receiver**: Receives files from the sender
122///
123/// Some protocol messages are role-specific (e.g., seed exchange direction).
124#[allow(dead_code)] // Part of protocol spec, not yet fully used
125#[derive(Debug, Clone, Copy, PartialEq, Eq)]
126pub enum Role {
127    /// This side is sending files
128    Sender,
129
130    /// This side is receiving files
131    Receiver,
132}
133
134impl Role {
135    /// Check if this is the sender role
136    #[allow(dead_code)]
137    #[must_use]
138    pub const fn is_sender(&self) -> bool {
139        matches!(self, Self::Sender)
140    }
141
142    /// Check if this is the receiver role
143    #[allow(dead_code)]
144    #[must_use]
145    pub const fn is_receiver(&self) -> bool {
146        matches!(self, Self::Receiver)
147    }
148
149    /// Get the opposite role
150    #[allow(dead_code)]
151    #[must_use]
152    pub const fn opposite(&self) -> Self {
153        match self {
154            Self::Sender => Self::Receiver,
155            Self::Receiver => Self::Sender,
156        }
157    }
158}

161// ChecksumSeed
162// ============================================================================
163
164/// Random seed for checksum algorithms
165///
166/// The checksum seed is used to randomize rolling checksums, which helps
167/// prevent collision attacks in untrusted environments. Both sides exchange
168/// seeds during the handshake if `XMIT_CHECKSUM_SEED` is negotiated.
169///
170/// # Wire Format
171///
172/// Seeds are transmitted as 4 bytes in little-endian format.
173#[derive(Debug, Clone, Copy, PartialEq, Eq)]
174pub struct ChecksumSeed {
175    /// The random seed value
176    pub seed: u32,
177}
178
179impl ChecksumSeed {
180    /// Generate a new random checksum seed
181    ///
182    /// # Example
183    ///
184    /// ```rust
185    /// use arsync::protocol::handshake::ChecksumSeed;
186    ///
187    /// let seed = ChecksumSeed::generate();
188    /// assert_ne!(seed.seed, 0); // Should be random
189    /// ```
190    #[must_use]
191    pub fn generate() -> Self {
192        use rand::Rng;
193        Self {
194            seed: rand::rng().random(),
195        }
196    }
197
198    /// Create a seed from raw bytes (little-endian)
199    ///
200    /// # Example
201    ///
202    /// ```rust
203    /// use arsync::protocol::handshake::ChecksumSeed;
204    ///
205    /// let bytes = [0x01, 0x02, 0x03, 0x04];
206    /// let seed = ChecksumSeed::from_bytes(bytes);
207    /// assert_eq!(seed.seed, 0x04030201);
208    /// ```
209    #[must_use]
210    pub const fn from_bytes(bytes: [u8; 4]) -> Self {
211        Self {
212            seed: u32::from_le_bytes(bytes),
213        }
214    }
215
216    /// Convert seed to bytes (little-endian)
217    ///
218    /// # Example
219    ///
220    /// ```rust
221    /// use arsync::protocol::handshake::ChecksumSeed;
222    ///
223    /// let seed = ChecksumSeed { seed: 0x04030201 };
224    /// let bytes = seed.to_bytes();
225    /// assert_eq!(bytes, [0x01, 0x02, 0x03, 0x04]);
226    /// ```
227    #[must_use]
228    #[allow(clippy::wrong_self_convention)] // to_bytes is standard for wire protocol serialization
229    pub const fn to_bytes(&self) -> [u8; 4] {
230        self.seed.to_le_bytes()
231    }
232
233    /// Check if the seed is zero (uninitialized)
234    #[allow(dead_code)]
235    #[must_use]
236    pub const fn is_zero(&self) -> bool {
237        self.seed == 0
238    }
239}

245/// Negotiated protocol capabilities
246///
247/// After the handshake, this structure contains the agreed-upon protocol
248/// version and feature flags that both sides support.
249///
250/// # Example
251///
252/// ```rust
253/// use arsync::protocol::handshake::{ProtocolCapabilities, XMIT_CHECKSUMS};
254///
255/// let caps = ProtocolCapabilities::new(31);
256/// assert_eq!(caps.version, 31);
257/// ```
258#[derive(Debug, Clone)]
259pub struct ProtocolCapabilities {
260    /// Negotiated protocol version (minimum of both sides)
261    pub version: u8,
262
263    /// Capability flags (intersection of both sides)
264    pub flags: u32,
265
266    /// Optional checksum seed (if `XMIT_CHECKSUM_SEED` is set)
267    pub checksum_seed: Option<u32>,
268}

270// Part of protocol spec, not yet fully used

353    /// Negotiate capabilities between client and server
354    ///
355    /// The negotiated capabilities are the intersection of what both sides support:
356    /// - Version: minimum of both versions
357    /// - Flags: bitwise AND of both flags
358    ///
359    /// # Example
360    ///
361    /// ```rust
362    /// use arsync::protocol::handshake::{ProtocolCapabilities, XMIT_CHECKSUMS, XMIT_SYMLINKS};
363    ///
364    /// let mut client = ProtocolCapabilities::new(31);
365    /// client.flags = XMIT_CHECKSUMS | XMIT_SYMLINKS;
366    ///
367    /// let mut server = ProtocolCapabilities::new(30);
368    /// server.flags = XMIT_CHECKSUMS; // Only checksums, no symlinks
369    ///
370    /// let negotiated = ProtocolCapabilities::negotiate(&client, &server);
371    /// assert_eq!(negotiated.version, 30); // Minimum version
372    /// assert!(negotiated.supports_checksums()); // Both support
373    /// assert!(!negotiated.supports_symlinks()); // Only client supports
374    /// ```
375    #[must_use]
376    pub fn negotiate(client: &Self, server: &Self) -> Self {
377        Self {
378            version: client.version.min(server.version),
379            flags: client.flags & server.flags, // Intersection
380            checksum_seed: None,                // Will be set during seed exchange
381        }
382    }

389/// State machine for protocol handshake
390///
391/// The handshake progresses through multiple states:
392///
393/// ```text
394/// Initial
395///   ↓
396/// VersionSent (sent our version)
397///   ↓
398/// VersionReceived (got remote version)
399///   ↓
400/// VersionNegotiated (computed effective version)
401///   ↓
402/// FlagsSent (sent our capabilities)
403///   ↓
404/// FlagsReceived (got remote capabilities)
405///   ↓
406/// CapabilitiesNegotiated (computed effective capabilities)
407///   ↓
408/// SeedExchange (exchanging checksum seeds, optional)
409///   ↓
410/// Complete (handshake done, ready for file transfer)
411/// ```
412///
413/// # Example
414///
415/// ```rust,no_run
416/// use arsync::protocol::handshake::{HandshakeState, Role};
417/// use arsync::protocol::pipe::PipeTransport;
418///
419/// # async fn example() -> anyhow::Result<()> {
420/// let mut transport = PipeTransport::from_stdio()?;
421/// let mut state = HandshakeState::Initial;
422///
423/// while !state.is_complete() {
424///     state = state.advance(&mut transport, Role::Sender).await?;
425/// }
426///
427/// let capabilities = state.get_capabilities().unwrap();
428/// println!("Handshake complete! Version: {}", capabilities.version);
429/// # Ok(())
430/// # }
431/// ```
432#[allow(dead_code)] // Part of protocol spec, not yet fully used
433#[derive(Debug, Clone)]
434pub enum HandshakeState {
435    /// Initial state - handshake not started
436    Initial,
437
438    /// Sent our protocol version to remote
439    VersionSent {
440        /// Our protocol version
441        our_version: u8,
442    },
443
444    /// Received remote protocol version
445    VersionReceived {
446        /// Our protocol version
447        our_version: u8,
448        /// Remote protocol version
449        remote_version: u8,
450    },
451
452    /// Negotiated effective protocol version
453    VersionNegotiated {
454        /// Effective protocol version (minimum of both)
455        protocol_version: u8,
456    },
457
458    /// Sent our capability flags to remote
459    FlagsSent {
460        /// Effective protocol version
461        protocol_version: u8,
462        /// Our capability flags
463        our_flags: u32,
464    },
465
466    /// Received remote capability flags
467    FlagsReceived {
468        /// Effective protocol version
469        protocol_version: u8,
470        /// Our capability flags
471        our_flags: u32,
472        /// Remote capability flags
473        remote_flags: u32,
474    },
475
476    /// Negotiated effective capabilities
477    CapabilitiesNegotiated {
478        /// Negotiated capabilities
479        capabilities: ProtocolCapabilities,
480    },
481
482    /// Exchanging checksum seeds (optional phase)
483    SeedExchange {
484        /// Capabilities with seed exchange in progress
485        capabilities: ProtocolCapabilities,
486    },
487
488    /// Handshake complete - ready for file transfer
489    Complete {
490        /// Final negotiated capabilities
491        capabilities: ProtocolCapabilities,
492        /// Checksum seed (if negotiated)
493        seed: Option<ChecksumSeed>,
494    },
495}

497// Part of protocol spec, not yet fully used
498#[allow(dead_code)]
499impl HandshakeState {
500    /// Check if handshake is complete
501    ///
502    /// # Example
503    ///
504    /// ```rust
505    /// use arsync::protocol::handshake::{HandshakeState, ProtocolCapabilities};
506    ///
507    /// let state = HandshakeState::Initial;
508    /// assert!(!state.is_complete());
509    ///
510    /// let state = HandshakeState::Complete {
511    ///     capabilities: ProtocolCapabilities::new(31),
512    ///     seed: None,
513    /// };
514    /// assert!(state.is_complete());
515    /// ```
516    #[must_use]
517    pub const fn is_complete(&self) -> bool {
518        matches!(self, Self::Complete { .. })
519    }
520
521    /// Get capabilities if handshake is complete
522    ///
523    /// Returns `Some(&ProtocolCapabilities)` if in Complete state, `None` otherwise.
524    #[must_use]
525    pub const fn get_capabilities(&self) -> Option<&ProtocolCapabilities> {
526        match self {
527            Self::Complete { capabilities, .. } => Some(capabilities),
528            _ => None,
529        }
530    }
531
532    /// Get checksum seed if handshake is complete and seed was negotiated
533    #[must_use]
534    pub const fn get_seed(&self) -> Option<ChecksumSeed> {
535        match self {
536            Self::Complete { seed, .. } => *seed,
537            _ => None,
538        }
539    }
540
541    /// Advance the handshake state machine
542    ///
543    /// This is the core state machine that implements the rsync handshake protocol.
544    /// Each call advances the state by one step, performing the necessary I/O and
545    /// validation for that step.
546    ///
547    /// # Errors
548    ///
549    /// Returns an error if:
550    /// - Transport I/O fails
551    /// - Protocol version is incompatible
552    /// - Invalid data is received
553    /// - Handshake is already complete
554    ///
555    /// # Example
556    ///
557    /// ```rust,no_run
558    /// # use arsync::protocol::handshake::{HandshakeState, Role};
559    /// # use arsync::protocol::pipe::PipeTransport;
560    /// # async fn example() -> anyhow::Result<()> {
561    /// let mut transport = PipeTransport::from_stdio()?;
562    /// let mut state = HandshakeState::Initial;
563    ///
564    /// while !state.is_complete() {
565    ///     state = state.advance(&mut transport, Role::Sender).await?;
566    /// }
567    /// # Ok(())
568    /// # }
569    /// ```
570    #[allow(clippy::cognitive_complexity)] // State machine complexity is inherent
571    #[allow(clippy::too_many_lines)] // State machine requires comprehensive handling
572    pub async fn advance<T>(self, transport: &mut T, role: Role) -> Result<Self>
573    where
574        T: Transport,
575    {
576        match self {
577            // ================================================================
578            // Initial → VersionSent: Send our protocol version
579            // ================================================================
580            Self::Initial => {
581                debug!("Handshake: Sending protocol version {}", PROTOCOL_VERSION);
582
583                write_all(transport, &[PROTOCOL_VERSION])
584                    .await
585                    .map_err(|e| anyhow::anyhow!("Failed to send protocol version: {e}"))?;
586
587                debug!("Handshake: Version sent");
588
589                Ok(Self::VersionSent {
590                    our_version: PROTOCOL_VERSION,
591                })
592            }
593
594            // ================================================================
595            // VersionSent → VersionReceived: Receive remote protocol version
596            // ================================================================
597            Self::VersionSent { our_version } => {
598                debug!("Handshake: Waiting for remote protocol version");
599
600                let mut buf = [0u8; 1];
601                read_exact(transport, &mut buf)
602                    .await
603                    .map_err(|e| anyhow::anyhow!("Failed to read remote protocol version: {e}"))?;
604
605                let remote_version = buf[0];
606
607                debug!(
608                    "Handshake: Received protocol version {} from remote",
609                    remote_version
610                );
611
612                // Validate remote version
613                if remote_version < MIN_PROTOCOL_VERSION {
614                    anyhow::bail!(
615                        "Unsupported remote protocol version: {remote_version} (minimum supported: {MIN_PROTOCOL_VERSION})"
616                    );
617                }
618
619                if remote_version > MAX_PROTOCOL_VERSION {
620                    warn!(
621                        "Remote protocol version {} is newer than maximum known version {}",
622                        remote_version, MAX_PROTOCOL_VERSION
623                    );
624                }
625
626                Ok(Self::VersionReceived {
627                    our_version,
628                    remote_version,
629                })
630            }
631
632            // ================================================================
633            // VersionReceived → VersionNegotiated: Compute effective version
634            // ================================================================
635            Self::VersionReceived {
636                our_version,
637                remote_version,
638            } => {
639                // Effective version is minimum of both
640                let protocol_version = our_version.min(remote_version);
641
642                info!(
643                    "Handshake: Protocol version negotiated: {} (our={}, remote={})",
644                    protocol_version, our_version, remote_version
645                );
646
647                Ok(Self::VersionNegotiated { protocol_version })
648            }
649
650            // ================================================================
651            // VersionNegotiated → FlagsSent: Send our capability flags
652            // ================================================================
653            Self::VersionNegotiated { protocol_version } => {
654                let our_flags = get_our_capabilities();
655
656                debug!("Handshake: Sending capability flags: 0x{:08X}", our_flags);
657
658                // Encode flags as varint
659                let mut buf = Vec::new();
660                crate::protocol::varint::encode_varint_into(u64::from(our_flags), &mut buf);
661
662                write_all(transport, &buf)
663                    .await
664                    .map_err(|e| anyhow::anyhow!("Failed to send capability flags: {e}"))?;
665
666                debug!("Handshake: Capability flags sent");
667
668                Ok(Self::FlagsSent {
669                    protocol_version,
670                    our_flags,
671                })
672            }
673
674            // ================================================================
675            // FlagsSent → FlagsReceived: Receive remote capability flags
676            // ================================================================
677            Self::FlagsSent {
678                protocol_version,
679                our_flags,
680            } => {
681                debug!("Handshake: Waiting for remote capability flags");
682
683                // Decode remote flags from varint
684                #[allow(clippy::cast_possible_truncation)]
685                // Capability flags are u32 by protocol spec
686                let remote_flags = crate::protocol::varint::decode_varint(transport)
687                    .await
688                    .map_err(|e| anyhow::anyhow!("Failed to read remote capability flags: {e}"))?
689                    as u32;
690
691                debug!(
692                    "Handshake: Received capability flags: 0x{:08X}",
693                    remote_flags
694                );
695
696                Ok(Self::FlagsReceived {
697                    protocol_version,
698                    our_flags,
699                    remote_flags,
700                })
701            }
702
703            // ================================================================
704            // FlagsReceived → CapabilitiesNegotiated: Compute effective capabilities
705            // ================================================================
706            Self::FlagsReceived {
707                protocol_version,
708                our_flags,
709                remote_flags,
710            } => {
711                // Negotiated capabilities are intersection of flags
712                let mut capabilities = ProtocolCapabilities::new(protocol_version);
713                capabilities.flags = our_flags & remote_flags;
714
715                info!(
716                    "Handshake: Capabilities negotiated: 0x{:08X} (our=0x{:08X}, remote=0x{:08X})",
717                    capabilities.flags, our_flags, remote_flags
718                );
719
720                debug!("  Checksums: {}", capabilities.supports_checksums());
721                debug!("  Hardlinks: {}", capabilities.supports_hardlinks());
722                debug!("  Symlinks: {}", capabilities.supports_symlinks());
723                debug!("  Devices: {}", capabilities.supports_devices());
724                debug!("  Xattrs: {}", capabilities.supports_xattrs());
725                debug!("  ACLs: {}", capabilities.supports_acls());
726                debug!("  Checksum seed: {}", capabilities.supports_checksum_seed());
727
728                Ok(Self::CapabilitiesNegotiated { capabilities })
729            }
730
731            // ================================================================
732            // CapabilitiesNegotiated → SeedExchange or Complete
733            // ================================================================
734            Self::CapabilitiesNegotiated { mut capabilities } => {
735                // Check if we need to exchange checksum seeds
736                if capabilities.supports_checksum_seed() {
737                    debug!("Handshake: Checksum seed exchange required");
738
739                    let seed = match role {
740                        Role::Sender => {
741                            // Sender generates and sends seed
742                            let seed = ChecksumSeed::generate();
743                            debug!("Handshake: Sending checksum seed: 0x{:08X}", seed.seed);
744
745                            let bytes = seed.to_bytes();
746                            write_all(transport, &bytes).await.map_err(|e| {
747                                anyhow::anyhow!("Failed to send checksum seed: {e}")
748                            })?;
749
750                            Some(seed)
751                        }
752                        Role::Receiver => {
753                            // Receiver reads seed from sender
754                            let mut bytes = [0u8; 4];
755                            read_exact(transport, &mut bytes).await.map_err(|e| {
756                                anyhow::anyhow!("Failed to receive checksum seed: {e}")
757                            })?;
758
759                            let seed = ChecksumSeed::from_bytes(bytes);
760                            debug!("Handshake: Received checksum seed: 0x{:08X}", seed.seed);
761
762                            Some(seed)
763                        }
764                    };
765
766                    capabilities.checksum_seed = seed.map(|s| s.seed);
767
768                    Ok(Self::SeedExchange { capabilities })
769                } else {
770                    debug!("Handshake: No checksum seed exchange needed");
771
772                    info!("Handshake: Complete (no seed exchange)");
773
774                    Ok(Self::Complete {
775                        capabilities,
776                        seed: None,
777                    })
778                }
779            }
780
781            // ================================================================
782            // SeedExchange → Complete
783            // ================================================================
784            Self::SeedExchange { capabilities } => {
785                let seed = capabilities.checksum_seed.map(|s| ChecksumSeed { seed: s });
786
787                info!("Handshake: Complete (with checksum seed)");
788
789                Ok(Self::Complete { capabilities, seed })
790            }
791
792            // ================================================================
793            // Complete: Terminal state
794            // ================================================================
795            Self::Complete { .. } => {
796                anyhow::bail!("Handshake already complete")
797            }
798        }
799    }
800}

806/// Perform handshake as sender
807///
808/// This is a convenience function that runs the complete handshake state machine
809/// in sender mode. The sender is the side that will be sending files.
810///
811/// # Errors
812///
813/// Returns an error if:
814/// - Transport I/O fails
815/// - Protocol version is incompatible
816/// - Handshake fails for any reason
817///
818/// # Example
819///
820/// ```rust,no_run
821/// use arsync::protocol::handshake::handshake_sender;
822/// use arsync::protocol::pipe::PipeTransport;
823///
824/// # async fn example() -> anyhow::Result<()> {
825/// let mut transport = PipeTransport::from_stdio()?;
826/// let capabilities = handshake_sender(&mut transport).await?;
827/// println!("Handshake complete! Version: {}", capabilities.version);
828/// # Ok(())
829/// # }
830/// ```
831#[allow(dead_code)] // Part of protocol spec, not yet fully used
832pub async fn handshake_sender<T: Transport>(transport: &mut T) -> Result<ProtocolCapabilities> {
833    info!("Starting handshake as sender");
834
835    let mut state = HandshakeState::Initial;
836
837    while !state.is_complete() {
838        state = state.advance(transport, Role::Sender).await?;
839    }
840
841    let capabilities = state
842        .get_capabilities()
843        .ok_or_else(|| anyhow::anyhow!("Handshake completed but capabilities not set"))?
844        .clone();
845
846    info!(
847        "Handshake complete (sender): version={}, flags=0x{:08X}",
848        capabilities.version, capabilities.flags
849    );
850
851    Ok(capabilities)
852}

902/// Perform handshake with specified role
903///
904/// This is the most general handshake function that accepts a role parameter.
905/// Use `handshake_sender()` or `handshake_receiver()` for more convenient APIs.
906///
907/// # Errors
908///
909/// Returns an error if:
910/// - Transport I/O fails
911/// - Protocol version is incompatible
912/// - Handshake fails for any reason
913///
914/// # Example
915///
916/// ```rust,no_run
917/// use arsync::protocol::handshake::{handshake, Role};
918/// use arsync::protocol::pipe::PipeTransport;
919///
920/// # async fn example() -> anyhow::Result<()> {
921/// let mut transport = PipeTransport::from_stdio()?;
922/// let role = Role::Sender;
923/// let capabilities = handshake(&mut transport, role).await?;
924/// println!("Handshake complete! Version: {}", capabilities.version);
925/// # Ok(())
926/// # }
927/// ```
928#[allow(dead_code)] // Part of protocol spec, not yet fully used
929pub async fn handshake<T: Transport>(
930    transport: &mut T,
931    role: Role,
932) -> Result<ProtocolCapabilities> {
933    match role {
934        Role::Sender => handshake_sender(transport).await,
935        Role::Receiver => handshake_receiver(transport).await,
936    }
937}

943/// Get our supported capabilities
944///
945/// Returns the capability flags that arsync supports. These represent features
946/// that arsync has ALREADY IMPLEMENTED locally:
947///
948/// - **Checksums**: Delta algorithm with rolling + strong checksums ✅
949/// - **Symlinks**: Copy symlinks as symlinks (-l/--links) ✅
950/// - **Hard links**: Track and preserve hard links (-H/--hard-links) ✅
951/// - **Devices**: Preserve device files and special files (-D/--devices) ✅
952/// - **Extended attributes**: Preserve xattrs (-X/--xattrs) ✅
953/// - **ACLs**: Preserve POSIX ACLs (-A/--acls) ✅
954/// - **Checksum seed**: Use random seed for security ✅
955/// - **Permissions**: Preserve file permissions (-p/--perms) ✅
956/// - **Timestamps**: Preserve modification times (-t/--times) ✅
957///
958/// Note: All these features are fully implemented in arsync's local sync code.
959/// The wire protocol just needs to transmit these attributes.
960#[must_use]
961pub const fn get_our_capabilities() -> u32 {
962    let mut flags = 0u32;
963
964    // What we support (arsync has LOCAL support for all of these!)
965    flags |= XMIT_CHECKSUMS; // ✅ Checksums
966    flags |= XMIT_SYMLINKS; // ✅ Symlinks (-l/--links)
967    flags |= XMIT_HARDLINKS; // ✅ Hard links (-H/--hard-links)
968    flags |= XMIT_DEVICES; // ✅ Device files (-D/--devices)
969    flags |= XMIT_XATTRS; // ✅ Extended attributes (-X/--xattrs)
970    flags |= XMIT_ACLS; // ✅ POSIX ACLs (-A/--acls)
971    flags |= XMIT_CHECKSUM_SEED; // ✅ Checksum seed
972    flags |= XMIT_PROTECTION; // ✅ Permissions (-p/--perms)
973    flags |= XMIT_TIMES; // ✅ Timestamps (-t/--times)
974                         // Note: We also support -U/--atimes and --crtimes locally!
975
976    // What we don't support in wire protocol yet
977    // (but may add in future)
978    // flags |= XMIT_SPARSE;      // Sparse file optimization (rsync-specific)
979
980    flags
981}
