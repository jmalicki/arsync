36/// rsync protocol uses multiplexed I/O with tags
37/// Tag values from rsync source code (io.c)
38const MPLEX_BASE: u8 = 7;
39
40/// Message tags (rsync protocol)
41#[derive(Debug, Clone, Copy, PartialEq, Eq)]
42#[repr(u8)]
43pub enum MessageTag {
44    Data = MPLEX_BASE,            // Regular data
45    ErrorXfer = MPLEX_BASE + 1,   // Transfer error
46    Info = MPLEX_BASE + 2,        // Info message
47    Error = MPLEX_BASE + 3,       // Error message
48    Warning = MPLEX_BASE + 4,     // Warning
49    ErrorSocket = MPLEX_BASE + 5, // Socket error
50    Log = MPLEX_BASE + 6,         // Log message
51    Client = MPLEX_BASE + 7,      // Client message
52    Redo = MPLEX_BASE + 9,        // Redo request
53    FList = MPLEX_BASE + 20,      // File list data
54    FName = MPLEX_BASE + 21,      // Filename
55    IoError = MPLEX_BASE + 22,    // I/O error
56    Success = MPLEX_BASE + 100,   // Success
57    NoSend = MPLEX_BASE + 101,    // Nothing to send
58}
59
60impl MessageTag {
61    /// Parse message tag from wire format byte
62    fn from_u8(tag: u8) -> Option<Self> {
63        match tag {
64            7 => Some(Self::Data),
65            8 => Some(Self::ErrorXfer),
66            9 => Some(Self::Info),
67            10 => Some(Self::Error),
68            11 => Some(Self::Warning),
69            12 => Some(Self::ErrorSocket),
70            13 => Some(Self::Log),
71            14 => Some(Self::Client),
72            16 => Some(Self::Redo),
73            27 => Some(Self::FList),
74            28 => Some(Self::FName),
75            29 => Some(Self::IoError),
76            107 => Some(Self::Success),
77            108 => Some(Self::NoSend),
78            _ => None,
79        }
80    }
81}

87    pub tag: MessageTag,
88    /// Message payload bytes
89    pub data: Vec<u8>,
90}

92/// Read a multiplexed message from rsync protocol stream
93pub async fn read_mplex_message<T: Transport>(transport: &mut T) -> Result<MultiplexMessage> {
94    // Read tag byte
95    let mut tag_buf = [0u8; 1];
96    transport::read_exact(transport, &mut tag_buf).await?;
97    let tag_byte = tag_buf[0];
98
99    // Parse tag
100    let tag = MessageTag::from_u8(tag_byte)
101        .ok_or_else(|| anyhow::anyhow!("Unknown rsync message tag: {}", tag_byte))?;
102
103    // Read length (3 bytes, little-endian, but only lower 24 bits used)
104    let mut len_buf = [0u8; 3];
105    transport::read_exact(transport, &mut len_buf).await?;
106    let length = u32::from_le_bytes([len_buf[0], len_buf[1], len_buf[2], 0]) as usize;
107
108    // Read data
109    let mut data = vec![0u8; length];
110    if length > 0 {
111        transport::read_exact(transport, &mut data).await?;
112    }
113
114    debug!("Read rsync message: tag={:?}, length={}", tag, length);
115
116    Ok(MultiplexMessage { tag, data })
117}

119/// Write a multiplexed message to rsync protocol stream
120pub async fn write_mplex_message<T: Transport>(
121    transport: &mut T,
122    tag: MessageTag,
123    data: &[u8],
124) -> Result<()> {
125    let length = data.len();
126
127    if length > 0xFFFFFF {
128        anyhow::bail!("Message too large for rsync protocol: {} bytes", length);
129    }
130
131    // Write tag
132    transport::write_all(transport, &[tag as u8]).await?;
133
134    // Write length (3 bytes, little-endian)
135    let len_bytes = (length as u32).to_le_bytes();
136    transport::write_all(transport, &len_bytes[0..3]).await?;
137
138    // Write data
139    if length > 0 {
140        transport::write_all(transport, data).await?;
141    }
142
143    debug!("Wrote rsync message: tag={:?}, length={}", tag, length);
144
145    Ok(())
146}

148/// Read a data message, handling multiplexed protocol
149pub async fn read_data<T: Transport>(transport: &mut T, buffer: &mut [u8]) -> Result<usize> {
150    loop {
151        let msg = read_mplex_message(transport).await?;
152
153        match msg.tag {
154            MessageTag::Data => {
155                // This is the data we want
156                let copy_len = msg.data.len().min(buffer.len());
157                buffer[..copy_len].copy_from_slice(&msg.data[..copy_len]);
158                return Ok(copy_len);
159            }
160            MessageTag::Info | MessageTag::Log => {
161                // Log messages, print to stderr
162                if let Ok(text) = String::from_utf8(msg.data) {
163                    debug!("rsync: {}", text.trim());
164                }
165            }
166            MessageTag::Error | MessageTag::ErrorXfer => {
167                // Error message
168                let error_msg = String::from_utf8_lossy(&msg.data);
169                anyhow::bail!("rsync error: {}", error_msg);
170            }
171            MessageTag::Warning => {
172                let warn_msg = String::from_utf8_lossy(&msg.data);
173                warn!("rsync warning: {}", warn_msg);
174            }
175            _ => {
176                debug!("Unexpected rsync message tag: {:?}", msg.tag);
177            }
178        }
179    }
180}

214    /// Read data message, skipping INFO/LOG messages
215    pub async fn read_data(&mut self, buf: &mut [u8]) -> Result<usize> {
216        loop {
217            let msg = self.read_message().await?;

219            match msg.tag {
220                MessageTag::Data => {
221                    let copy_len = msg.data.len().min(buf.len());
222                    buf[..copy_len].copy_from_slice(&msg.data[..copy_len]);
223                    return Ok(copy_len);
224                }
225                MessageTag::Info | MessageTag::Log => {
226                    if let Ok(text) = String::from_utf8(msg.data) {
227                        debug!("rsync: {}", text.trim());
228                    }
229                }
230                MessageTag::Error | MessageTag::ErrorXfer => {
231                    let error_msg = String::from_utf8_lossy(&msg.data);
232                    anyhow::bail!("rsync error: {}", error_msg);
233                }
234                MessageTag::Warning => {
235                    let warn_msg = String::from_utf8_lossy(&msg.data);
236                    warn!("rsync warning: {}", warn_msg);
237                }
238                _ => {
239                    debug!("Ignoring rsync message tag: {:?}", msg.tag);
240                }
241            }
242        }
243    }
244
245    /// Expect a specific message tag
246    pub async fn expect_tag(&mut self, expected: MessageTag) -> Result<Vec<u8>> {
247        let msg = self.read_message().await?;
248
249        if msg.tag == expected {
250            Ok(msg.data)
251        } else {
252            anyhow::bail!("Expected tag {:?}, got {:?}", expected, msg.tag);
253        }
254    }
255
256    /// Read multiple messages until end marker
257    pub async fn read_until_empty(&mut self, expected_tag: MessageTag) -> Result<Vec<Vec<u8>>> {
258        let mut messages = Vec::new();
259
260        loop {
261            let msg = self.read_message().await?;
262
263            if msg.tag != expected_tag {
264                anyhow::bail!("Expected tag {:?}, got {:?}", expected_tag, msg.tag);
265            }
266
267            if msg.data.is_empty() {
268                // Empty message = end marker
269                break;
270            }
271
272            messages.push(msg.data);
273        }
274
275        Ok(messages)
276    }

368/// rsync file list flags (from flist.c)
369mod file_flags {
370    /// Top-level directory flag
371    #[allow(dead_code)]
372    pub const XMIT_TOP_DIR: u8 = 0x01;
373    /// Mode unchanged from previous file (delta encoding)
374    #[allow(dead_code)]
375    pub const XMIT_SAME_MODE: u8 = 0x02;
376    /// Extended flags follow this byte
377    #[allow(dead_code)]
378    pub const XMIT_EXTENDED_FLAGS: u8 = 0x04;
379    /// UID unchanged from previous file (delta encoding)
380    #[allow(dead_code)]
381    pub const XMIT_SAME_UID: u8 = 0x10;
382    /// GID unchanged from previous file (delta encoding)
383    #[allow(dead_code)]
384    pub const XMIT_SAME_GID: u8 = 0x20;
385    /// Name matches previous file (hardlink)
386    #[allow(dead_code)]
387    pub const XMIT_SAME_NAME: u8 = 0x40;
388    /// Filename is longer than 255 bytes
389    pub const XMIT_LONG_NAME: u8 = 0x80;
390}

392/// Encode file list in rsync wire format (simplified - no delta encoding yet)
393///
394/// This sends each file as MSG_FLIST tagged message with varint-encoded fields.
395/// Simplified version: no mtime deltas, no directory grouping.
396pub async fn encode_file_list_rsync<T: Transport>(
397    writer: &mut MultiplexWriter<T>,
398    files: &[FileEntry],
399) -> Result<()> {
400    debug!("Encoding {} files in rsync format", files.len());
401
402    for file in files {
403        let mut entry = Vec::new();
404
405        // Flags byte
406        let mut flags = 0u8;
407        if file.path.len() > 255 {
408            flags |= file_flags::XMIT_LONG_NAME;
409        }
410        // For now, no delta encoding, so no SAME_MODE/SAME_UID/SAME_GID flags
411
412        entry.push(flags);
413
414        // Path (varint length + bytes)
415        encode_varint_into(file.path.len() as u64, &mut entry);
416        entry.extend(file.path.as_bytes());
417
418        // File size (varint)
419        encode_varint_into(file.size, &mut entry);
420
421        // Mtime (varint, absolute - no delta encoding yet)
422        // rsync uses unsigned for mtime, we need to convert
423        let mtime_unsigned = if file.mtime < 0 {
424            0u64 // Clamp negative times to 0
425        } else {
426            file.mtime as u64
427        };
428        encode_varint_into(mtime_unsigned, &mut entry);
429
430        // Mode (varint)
431        encode_varint_into(file.mode as u64, &mut entry);
432
433        // uid/gid (varint)
434        encode_varint_into(file.uid as u64, &mut entry);
435        encode_varint_into(file.gid as u64, &mut entry);
436
437        // Symlink target if applicable
438        if file.is_symlink {
439            if let Some(ref target) = file.symlink_target {
440                encode_varint_into(target.len() as u64, &mut entry);
441                entry.extend(target.as_bytes());
442            }
443        }
444
445        // Send as MSG_FLIST tagged message
446        writer.write_message(MessageTag::FList, &entry).await?;
447
448        debug!("Encoded file: {} ({} bytes)", file.path, entry.len());
449    }
450
451    // End-of-list marker (empty MSG_FLIST)
452    writer.write_message(MessageTag::FList, &[]).await?;
453    debug!("Sent end-of-list marker");
454
455    Ok(())
456}

458/// Decode file list from rsync wire format
459///
460/// Reads MSG_FLIST tagged messages until empty message (end marker).
461pub async fn decode_file_list_rsync<T: Transport>(
462    reader: &mut MultiplexReader<T>,
463) -> Result<Vec<FileEntry>> {
464    debug!("Decoding file list in rsync format");
465
466    let mut files = Vec::new();
467
468    loop {
469        // Read next file list message
470        let msg = reader.read_message().await?;
471
472        if msg.tag != MessageTag::FList {
473            anyhow::bail!("Expected MSG_FLIST, got {:?}", msg.tag);
474        }
475
476        if msg.data.is_empty() {
477            // End of list marker
478            debug!("Received end-of-list marker");
479            break;
480        }
481
482        // Decode file entry from message data
483        let file = decode_file_entry(&msg.data)?;
484        debug!("Decoded file: {} ({} bytes)", file.path, file.size);
485        files.push(file);
486    }
487
488    debug!("Decoded {} files total", files.len());
489    Ok(files)
490}

492/// Decode a single file entry from bytes
493pub fn decode_file_entry(data: &[u8]) -> Result<FileEntry> {
494    let mut cursor = Cursor::new(data);
495
496    // Flags byte
497    let mut flags_buf = [0u8; 1];
498    std::io::Read::read_exact(&mut cursor, &mut flags_buf)?;
499    let _flags = flags_buf[0];
500
501    // Path length and path
502    let path_len = decode_varint_sync(&mut cursor)? as usize;
503    let mut path_buf = vec![0u8; path_len];
504    std::io::Read::read_exact(&mut cursor, &mut path_buf)?;
505    let path = String::from_utf8(path_buf)?;
506
507    // File size
508    let size = decode_varint_sync(&mut cursor)?;
509
510    // Mtime
511    let mtime = decode_varint_sync(&mut cursor)? as i64;
512
513    // Mode
514    let mode = decode_varint_sync(&mut cursor)? as u32;
515
516    // uid/gid
517    let uid = decode_varint_sync(&mut cursor)? as u32;
518    let gid = decode_varint_sync(&mut cursor)? as u32;
519
520    // Check if symlink (we need to infer from mode bits)
521    let is_symlink = (mode & 0o170000) == 0o120000; // S_IFLNK
522
523    // Symlink target if applicable
524    let symlink_target = if is_symlink {
525        let target_len = decode_varint_sync(&mut cursor)? as usize;
526        let mut target_buf = vec![0u8; target_len];
527        std::io::Read::read_exact(&mut cursor, &mut target_buf)?;
528        Some(String::from_utf8(target_buf)?)
529    } else {
530        None
531    };
532
533    Ok(FileEntry {
534        path,
535        size,
536        mtime,
537        mode,
538        uid,
539        gid,
540        is_symlink,
541        symlink_target,
542    })
543}

545/// Decode varint from synchronous reader (for use with Cursor)
546fn decode_varint_sync(reader: &mut Cursor<&[u8]>) -> Result<u64> {
547    use std::io::Read;
548
549    let mut result = 0u64;
550    let mut shift = 0;
551
552    loop {
553        let mut byte_buf = [0u8; 1];
554        reader.read_exact(&mut byte_buf)?;
555        let byte = byte_buf[0];
556
557        result |= ((byte & 0x7F) as u64) << shift;
558
559        if (byte & 0x80) == 0 {
560            break;
561        }
562
563        shift += 7;
564
565        if shift > 63 {
566            anyhow::bail!("Varint overflow");
567        }
568    }
569
570    Ok(result)
571}

579/// Block checksum in rsync wire format
580#[derive(Debug, Clone)]
581pub struct RsyncBlockChecksum {
582    /// Rolling checksum (32-bit, fast but collision-prone)
583    pub weak: u32,
584    /// Strong checksum (MD5 or MD4, variable length: 2 or 16 bytes)
585    pub strong: Vec<u8>,
586}
587
588/// Send block checksums in rsync wire format
589///
590/// rsync format:
591/// - Header: \[block_count\]\[block_size\]\[remainder\]\[checksum2_length\]
592/// - Then each block: \[weak\]\[strong\] (no offset/index - implicit!)
593///
594/// # Errors
595///
596/// Returns an error if writing to the transport fails.
597pub async fn send_block_checksums_rsync<T: Transport>(
598    writer: &mut MultiplexWriter<T>,
599    data: &[u8],
600    block_size: usize,
601    seed: u32,
602) -> Result<()> {
603    if data.is_empty() {
604        // Send empty checksum list
605        let header = vec![
606            0u8, 0, 0, 0, // block_count = 0
607            0, 0, 0, 0, // block_size = 0
608            0, 0, 0, 0, // remainder = 0
609            16, 0, 0, 0, // checksum2_length = 16 (MD5)
610        ];
611        writer.write_message(MessageTag::Data, &header).await?;
612        return Ok(());
613    }
614
615    let block_count = data.len().div_ceil(block_size);
616    let remainder = data.len() % block_size;
617    let checksum2_length = 16u32; // MD5 = 16 bytes
618
619    // Build header
620    let mut message = Vec::new();
621    message.extend((block_count as u32).to_le_bytes());
622    message.extend((block_size as u32).to_le_bytes());
623    message.extend((remainder as u32).to_le_bytes());
624    message.extend(checksum2_length.to_le_bytes());
625
626    // Generate and append checksums
627    let mut offset = 0;
628    while offset < data.len() {
629        let end = (offset + block_size).min(data.len());
630        let block = &data[offset..end];
631
632        // Compute checksums with seed
633        let weak = rolling_checksum_with_seed(block, seed);
634        let strong = strong_checksum(block);
635
636        message.extend(weak.to_le_bytes());
637        message.extend(strong);
638
639        offset = end;
640    }
641
642    // Send as MSG_DATA
643    writer.write_message(MessageTag::Data, &message).await?;
644    debug!(
645        "Sent {} block checksums (block_size={}, seed={})",
646        block_count, block_size, seed
647    );
648
649    Ok(())
650}

652/// Receive block checksums in rsync wire format
653///
654/// # Errors
655///
656/// Returns an error if:
657/// - Reading from the transport fails
658/// - Invalid message tag received (expected MSG_DATA)
659/// - Malformed checksum data
660pub async fn receive_block_checksums_rsync<T: Transport>(
661    reader: &mut MultiplexReader<T>,
662) -> Result<(Vec<RsyncBlockChecksum>, usize)> {
663    // Read MSG_DATA containing checksums
664    let msg = reader.read_message().await?;
665
666    if msg.tag != MessageTag::Data {
667        anyhow::bail!("Expected MSG_DATA for checksums, got {:?}", msg.tag);
668    }
669
670    if msg.data.len() < 16 {
671        // Empty or invalid checksum list
672        return Ok((vec![], 0));
673    }
674
675    let mut cursor = Cursor::new(&msg.data[..]);
676
677    // Read header
678    let mut buf4 = [0u8; 4];
679
680    std::io::Read::read_exact(&mut cursor, &mut buf4)?;
681    let block_count = u32::from_le_bytes(buf4) as usize;
682
683    std::io::Read::read_exact(&mut cursor, &mut buf4)?;
684    let block_size = u32::from_le_bytes(buf4) as usize;
685
686    std::io::Read::read_exact(&mut cursor, &mut buf4)?;
687    let _remainder = u32::from_le_bytes(buf4);
688
689    std::io::Read::read_exact(&mut cursor, &mut buf4)?;
690    let checksum2_length = u32::from_le_bytes(buf4) as usize;
691
692    debug!(
693        "Receiving {} checksums (block_size={}, strong_len={})",
694        block_count, block_size, checksum2_length
695    );
696
697    // Read checksums
698    let mut checksums = Vec::with_capacity(block_count);
699    for _ in 0..block_count {
700        std::io::Read::read_exact(&mut cursor, &mut buf4)?;
701        let weak = u32::from_le_bytes(buf4);
702
703        let mut strong = vec![0u8; checksum2_length];
704        std::io::Read::read_exact(&mut cursor, &mut strong)?;
705
706        checksums.push(RsyncBlockChecksum { weak, strong });
707    }
708
709    Ok((checksums, block_size))
710}

718/// Convert delta instructions to rsync token stream
719///
720/// rsync token format:
721/// - 0: End of data marker
722/// - 1-96: Literal run (N bytes of data follow)
723/// - 97-255: Block match with offset encoding
724#[must_use]
725pub fn delta_to_tokens(delta: &[DeltaInstruction]) -> Vec<u8> {
726    let mut tokens = Vec::new();
727    let mut last_block_index: i64 = -1;
728
729    for instruction in delta {
730        match instruction {
731            DeltaInstruction::Literal(data) => {
732                // Split into chunks of max 96 bytes
733                for chunk in data.chunks(96) {
734                    let token = chunk.len() as u8; // 1-96
735                    tokens.push(token);
736                    tokens.extend_from_slice(chunk);
737                }
738            }
739            DeltaInstruction::BlockMatch {
740                block_index,
741                length: _,
742            } => {
743                // Calculate offset from last block
744                let offset = if last_block_index < 0 {
745                    *block_index as i64
746                } else {
747                    (*block_index as i64) - (last_block_index + 1)
748                };
749
750                // Encode as token
751                if offset < 0 {
752                    // Shouldn't happen, but handle it
753                    tokens.push(97); // Offset 0
754                } else if offset < 16 {
755                    // Simple encoding: 97 + offset
756                    tokens.push(97 + offset as u8);
757                } else {
758                    // Complex encoding for large offsets
759                    // For now, use simplified version
760                    // TODO: Implement full rsync offset encoding
761                    let bit_count = 64 - (offset as u64).leading_zeros();
762                    let token = 97 + ((bit_count as u8) << 4);
763                    tokens.push(token);
764
765                    // Send offset bytes (simplified)
766                    tokens.extend((offset as u32).to_le_bytes());
767                }
768
769                last_block_index = *block_index as i64;
770            }
771        }
772    }
773
774    // End of data marker
775    tokens.push(0);
776
777    tokens
778}

780/// Parse rsync token stream back to delta instructions
781///
782/// # Errors
783///
784/// Returns an error if:
785/// - Literal data is truncated (insufficient bytes in token stream)
786/// - Block offset data is truncated
787pub fn tokens_to_delta(
788    tokens: &[u8],
789    block_checksums: &[RsyncBlockChecksum],
790) -> anyhow::Result<Vec<DeltaInstruction>> {
791    let mut instructions = Vec::new();
792    let mut pos = 0;
793    let mut last_block_index: i64 = -1;
794
795    while pos < tokens.len() {
796        let token = tokens[pos];
797        pos += 1;
798
799        if token == 0 {
800            // End of data
801            break;
802        } else if token <= 96 {
803            // Literal run
804            let literal_len = token as usize;
805            if pos + literal_len > tokens.len() {
806                anyhow::bail!("Literal data truncated");
807            }
808
809            let literal_data = tokens[pos..pos + literal_len].to_vec();
810            pos += literal_len;
811
812            instructions.push(DeltaInstruction::Literal(literal_data));
813        } else {
814            // Block match (97-255)
815            let offset = if token < 113 {
816                // Simple encoding: offset = token - 97
817                (token - 97) as i64
818            } else {
819                // Complex encoding
820                let _bit_count = ((token - 97) >> 4) as usize;
821
822                if pos + 4 > tokens.len() {
823                    anyhow::bail!("Block offset truncated");
824                }
825
826                let mut offset_bytes = [0u8; 4];
827                offset_bytes.copy_from_slice(&tokens[pos..pos + 4]);
828                pos += 4;
829
830                u32::from_le_bytes(offset_bytes) as i64
831            };
832
833            // Calculate absolute block index
834            let block_index = if last_block_index < 0 {
835                offset as u32
836            } else {
837                ((last_block_index + 1) + offset) as u32
838            };
839
840            // Get block length from checksums
841            let length = if (block_index as usize) < block_checksums.len() {
842                block_checksums[block_index as usize].strong.len() as u32
843            } else {
844                4096 // Default block size if not in checksums
845            };
846
847            instructions.push(DeltaInstruction::BlockMatch {
848                block_index,
849                length,
850            });
851
852            last_block_index = block_index as i64;
853        }
854    }
855
856    Ok(instructions)
857}

859/// Send delta in rsync token format
860///
861/// # Errors
862///
863/// Returns an error if writing to the transport fails.
864pub async fn send_delta_rsync<T: Transport>(
865    writer: &mut MultiplexWriter<T>,
866    delta: &[DeltaInstruction],
867) -> Result<()> {
868    let tokens = delta_to_tokens(delta);
869    writer.write_message(MessageTag::Data, &tokens).await?;
870    debug!("Sent {} bytes of delta tokens", tokens.len());
871    Ok(())
872}

874/// Receive delta in rsync token format
875///
876/// # Errors
877///
878/// Returns an error if:
879/// - Reading from the transport fails
880/// - Invalid message tag received (expected MSG_DATA)
881/// - Token parsing fails (malformed delta)
882pub async fn receive_delta_rsync<T: Transport>(
883    reader: &mut MultiplexReader<T>,
884    checksums: &[RsyncBlockChecksum],
885) -> Result<Vec<DeltaInstruction>> {
886    let msg = reader.read_message().await?;
887
888    if msg.tag != MessageTag::Data {
889        anyhow::bail!("Expected MSG_DATA for delta, got {:?}", msg.tag);
890    }
891
892    tokens_to_delta(&msg.data, checksums)
893}

901///
902/// Generic over any Transport implementation (SSH, pipes, etc.)

904/// # Errors
905///
906/// Returns an error if:
907/// - Transport operations fail
908/// - File list decoding fails
909/// - File system operations fail (creating directories/files)
910pub async fn rsync_receive<T: Transport>(
911    _args: &Args,
912    transport: T,
913    dest_path: &Path,
914) -> Result<SyncStats> {
915    let start = Instant::now();
916    info!("rsync-compat receiver: Starting (file list only)");
917
918    // Wrap transport in multiplex reader
919    let mut reader = MultiplexReader::new(transport);
920
921    // Phase 1: Handshake (simplified - just version, no seed yet)
922    debug!("rsync-compat: Reading version from rsync sender");
923    // rsync sends version as raw byte (NOT tagged)
924    // We need to read it directly from transport, not via multiplex
925    // TODO: This is a challenge - need access to raw transport
926
927    // Phase 2: Receive file list
928    info!("rsync-compat: Receiving file list from rsync");
929    let files = decode_file_list_rsync(&mut reader).await?;
930    info!("rsync-compat: Received {} files from rsync", files.len());
931
932    // Print file list for validation
933    for file in &files {
934        info!(
935            "  File: {} ({} bytes, mode {:o})",
936            file.path, file.size, file.mode
937        );
938    }
939
940    // Phase 3: For now, just create empty files (no actual transfer yet)
941    info!("rsync-compat: Creating file structure (no content yet)");
942    for file in &files {
943        let file_path = dest_path.join(&file.path);
944
945        if let Some(parent) = file_path.parent() {
946            fs::create_dir_all(parent)?;
947        }
948
949        if file.is_symlink {
950            if let Some(ref target) = file.symlink_target {
951                #[cfg(unix)]
952                {
953                    use std::os::unix::fs as unix_fs;
954                    unix_fs::symlink(target, &file_path)?;
955                }
956            }
957        } else {
958            // Create empty placeholder
959            fs::write(&file_path, b"")?;
960        }
961    }
962
963    info!("rsync-compat: File list exchange complete (content transfer not implemented)");
964
965    Ok(SyncStats {
966        files_copied: files.len() as u64,
967        bytes_copied: 0, // No actual content transferred yet
968        duration: start.elapsed(),
969    })
970}
