1//! Adaptive concurrency control with file descriptor awareness
2//!
3//! This module provides self-adaptive concurrency control that automatically
4//! adjusts the number of concurrent operations based on resource availability,
5//! particularly file descriptor exhaustion.
6//!
7//! # Architecture
8//!
9//! Each module owns its configuration:
10//! - `ConcurrencyOptions` - Configuration for concurrency control (owned by this module)
11//! - `AdaptiveConcurrencyController` - Runtime controller that uses the options
12

20/// Type alias for a shared semaphore wrapped in `Arc`

23/// `Semaphore` in `Arc` when sharing across tasks.
24type SharedSemaphore = Arc<Semaphore>;

31/// Concurrency control configuration options
32///
33/// This struct contains all configuration for the adaptive concurrency controller.
34/// It's owned by this module and can be created from CLI args.
35///
36/// Uses `NonZeroUsize` to guarantee at compile-time that `max_files_in_flight` >= 1.
37#[derive(Debug, Clone)]
38pub struct ConcurrencyOptions {
39    /// Maximum number of concurrent operations (guaranteed >= 1)
40    max_files_in_flight: NonZeroUsize,
41    /// Minimum permits to maintain when adapting (never go below this, guaranteed >= 1)
42    min_permits: NonZeroUsize,
43    /// If true, fail immediately on resource exhaustion; if false, adapt automatically
44    fail_on_exhaustion: bool,
45}
46
47impl ConcurrencyOptions {
48    /// Create new concurrency options with validation
49    ///
50    /// # Arguments
51    ///
52    /// * `max_files_in_flight` - Maximum concurrent operations (will be clamped to >= 1)
53    /// * `fail_on_exhaustion` - If true, fail on EMFILE; if false, adapt
54    ///
55    /// # Panics
56    ///
57    /// This function will panic if `max_files_in_flight` is 0 in debug builds.
58    /// In release builds, it will be clamped to 1.
59    #[must_use]
60    pub fn new(max_files_in_flight: usize, fail_on_exhaustion: bool) -> Self {
61        // Ensure max_files_in_flight is at least 1
62        let max_files_in_flight = NonZeroUsize::new(max_files_in_flight).unwrap_or_else(|| {
63            debug_assert!(false, "max_files_in_flight must be >= 1");
64            // SAFETY: 1 is non-zero
65            unsafe { NonZeroUsize::new_unchecked(1) }
66        });
67
68        // Compute minimum permits: never go below 10 or 10% of max
69        let min_value = std::cmp::max(10, max_files_in_flight.get() / 10);
70        // SAFETY: min_value is at least 10, which is non-zero
71        let min_permits = unsafe { NonZeroUsize::new_unchecked(min_value) };
72
73        Self {
74            max_files_in_flight,
75            min_permits,
76            fail_on_exhaustion,
77        }
78    }
79
80    /// Get the maximum files in flight
81    #[must_use]
82    pub const fn max_files_in_flight(&self) -> usize {
83        self.max_files_in_flight.get()
84    }
85
86    /// Get the minimum permits (floor for adaptive reduction)
87    #[must_use]
88    pub const fn min_permits(&self) -> usize {
89        self.min_permits.get()
90    }
91
92    /// Check if should fail on exhaustion
93    #[must_use]
94    pub const fn fail_on_exhaustion(&self) -> bool {
95        self.fail_on_exhaustion
96    }
97}

103/// Adaptive concurrency controller that responds to resource constraints
104///
105/// This wraps a semaphore and automatically reduces concurrency when
106/// file descriptor exhaustion (EMFILE) is detected, then gradually
107/// increases it again when resources are available.
108#[derive(Clone)]
109#[allow(dead_code)]
110pub struct AdaptiveConcurrencyController {
111    /// The underlying semaphore
112    semaphore: SharedSemaphore,
113    /// Counter for EMFILE errors
114    emfile_errors: Arc<AtomicUsize>,
115    /// Flag indicating if we've already warned about FD exhaustion
116    emfile_warned: Arc<AtomicBool>,
117    /// Minimum permits to maintain
118    min_permits: usize,
119    /// Whether to fail hard on exhaustion (true = fail, false = adapt)
120    fail_on_exhaustion: bool,
121}
122
123#[allow(dead_code)]
124impl AdaptiveConcurrencyController {
125    /// Create a new adaptive controller from options
126    ///
127    /// # Arguments
128    ///
129    /// * `options` - Concurrency configuration options (validated and ready to use)
130    #[must_use]
131    pub fn new(options: &ConcurrencyOptions) -> Self {
132        Self {
133            semaphore: Arc::new(Semaphore::new(options.max_files_in_flight())),
134            emfile_errors: Arc::new(AtomicUsize::new(0)),
135            emfile_warned: Arc::new(AtomicBool::new(false)),
136            min_permits: options.min_permits(),
137            fail_on_exhaustion: options.fail_on_exhaustion(),
138        }
139    }
140
141    /// Acquire a permit
142    pub async fn acquire(&self) -> compio_sync::SemaphorePermit<'_> {
143        self.semaphore.acquire().await
144    }
145
146    /// Handle an error, checking if it's EMFILE and adapting or failing as configured
147    ///
148    /// If this is an EMFILE error:
149    /// - If `fail_on_exhaustion` is true: Returns error immediately
150    /// - If `fail_on_exhaustion` is false: Adapts concurrency and returns Ok
151    /// - If not an EMFILE error: Returns Ok
152    ///
153    /// # Errors
154    ///
155    /// Returns `FdExhaustion` error if EMFILE detected and `fail_on_exhaustion` is true
156    pub fn handle_error(&self, error: &SyncError) -> crate::error::Result<()> {
157        use crate::error::SyncError;
158
159        // Check if this is a file descriptor exhaustion error
160        if Self::is_emfile_error(error) {
161            let count = self.emfile_errors.fetch_add(1, Ordering::Relaxed) + 1;
162
163            // Check if we should fail hard
164            if self.fail_on_exhaustion {
165                return Err(SyncError::FdExhaustion(format!(
166                    "File descriptor exhaustion detected (--no-adaptive-concurrency is set). \
167                     Error: {error}. \
168                     Either increase ulimit or remove --no-adaptive-concurrency flag."
169                )));
170            }
171
172            // Otherwise, adapt every N errors to avoid over-reaction
173            if count % 5 == 1 {
174                self.adapt_to_fd_exhaustion();
175            }
176        }
177        Ok(())
178    }
179
180    /// Check if an error is EMFILE (too many open files)
181    fn is_emfile_error(error: &SyncError) -> bool {
182        let error_str = format!("{error:?}");
183        error_str.contains("Too many open files")
184            || error_str.contains("EMFILE")
185            || error_str.contains("os error 24")
186    }
187
188    /// Adapt to file descriptor exhaustion by reducing concurrency
189    fn adapt_to_fd_exhaustion(&self) {
190        let current_available = self.semaphore.available_permits();
191        let current_max = self.semaphore.max_permits();
192
193        // Reduce by 25% or minimum 10, but never go below min_permits
194        let reduction = std::cmp::max(10, current_max / 4);
195        let actual_reduced = self.semaphore.reduce_permits(reduction);
196
197        let new_max = current_max - actual_reduced;
198
199        if actual_reduced > 0 {
200            if self.emfile_warned.swap(true, Ordering::Relaxed) {
201                // Subsequent reductions - be brief
202                warn!(
203                    "Reducing concurrency further due to FD exhaustion: {} → {} (-{})",
204                    current_max, new_max, actual_reduced
205                );
206            } else {
207                // First time warning - be verbose
208                warn!(
209                    "⚠️  FILE DESCRIPTOR EXHAUSTION DETECTED (EMFILE)\n\
210                     \n\
211                     arsync has hit the system file descriptor limit.\n\
212                     \n\
213                     Self-adaptive response:\n\
214                     - Reduced concurrent operations: {} → {} (-{})\n\
215                     - Currently available: {}\n\
216                     - Minimum limit: {}\n\
217                     \n\
218                     This may slow down processing but prevents crashes.\n\
219                     \n\
220                     To avoid this:\n\
221                     - Increase ulimit: ulimit -n 100000\n\
222                     - Or use --max-files-in-flight to set lower initial concurrency\n\
223                     \n\
224                     Continuing with reduced concurrency...",
225                    current_max, new_max, actual_reduced, current_available, self.min_permits
226                );
227            }
228        }
229    }
230
231    /// Get current statistics
232    #[must_use]
233    #[allow(dead_code)] // Public API for future monitoring/metrics
234    pub fn stats(&self) -> ConcurrencyStats {
235        ConcurrencyStats {
236            max_permits: self.semaphore.max_permits(),
237            available_permits: self.semaphore.available_permits(),
238            in_use: self.semaphore.max_permits() - self.semaphore.available_permits(),
239            emfile_errors: self.emfile_errors.load(Ordering::Relaxed),
240        }
241    }
242}

258/// Check system file descriptor limits and warn if too low
259///
260/// Returns the soft limit for file descriptors
261///
262/// # Errors
263///
264/// Returns an error if getrlimit system call fails
265pub fn check_fd_limits() -> std::io::Result<u64> {
266    use libc::{getrlimit, rlimit, RLIMIT_NOFILE};
267
268    unsafe {
269        let mut limit = rlimit {
270            rlim_cur: 0,
271            rlim_max: 0,
272        };
273
274        if getrlimit(RLIMIT_NOFILE, &raw mut limit) == 0 {
275            let soft_limit = limit.rlim_cur;
276
277            // Warn if limit seems low for high-performance operations
278            if soft_limit < 10000 {
279                warn!(
280                    "⚠️  File descriptor limit is low: {}\n\
281                     \n\
282                     For optimal performance with arsync, consider:\n\
283                     ulimit -n 100000\n\
284                     \n\
285                     Current limit may cause FD exhaustion on large operations.\n\
286                     arsync will adapt automatically if this occurs.",
287                    soft_limit
288                );
289            } else {
290                tracing::info!("File descriptor limit: {} (adequate)", soft_limit);
291            }
292
293            Ok(soft_limit)
294        } else {
295            Err(std::io::Error::last_os_error())
296        }
297    }
298}
