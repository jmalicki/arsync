1//! Hardlink tracking for race-free hardlink synchronization
2//!
3//! This module provides `FilesystemTracker` which uses condition variables to ensure
4//! that hardlinked files are handled correctly in concurrent scenarios:
5//! - The first task to encounter a hardlinked inode becomes the "copier"
6//! - Subsequent tasks become "linkers" that wait for the copier to finish
7//! - The copier signals completion, allowing linkers to create hardlinks
8
9use dashmap::DashMap;
10use std::os::unix::fs::MetadataExt;
11use std::path::{Path, PathBuf};
12use std::sync::atomic::{AtomicU64, Ordering};
13use std::sync::Arc;
14use tracing::debug;
15
16/// Inode information used as a key for hardlink tracking
17#[derive(Debug, Clone, Copy, Hash, PartialEq, Eq)]
18pub struct InodeInfo {
19    /// Device ID
20    pub dev: u64,
21    /// Inode number
22    pub ino: u64,
23}
24
25/// Hardlink tracking information with concurrent synchronization
26///
27/// Uses `Condvar` to ensure race-free hardlink creation:
28/// - First task to register becomes the "copier" (stores `dst_path`, returns immediately)
29/// - Subsequent tasks are "linkers" (wait on condvar inside `register_file`, then return)
30/// - Copier signals condvar when copy completes (success or failure)
31/// - Linkers wake up and naturally succeed/fail based on whether dst file exists
32pub struct HardlinkInfo {
33    /// Original file path (immutable after creation)
34    #[allow(dead_code)]
35    pub original_path: std::path::PathBuf,
36    /// Inode number (immutable after creation)
37    pub inode_number: u64,
38    /// Number of hardlinks found (incremented atomically)
39    pub link_count: AtomicU64,
40    /// Destination path (set at registration time by first copier)
41    pub dst_path: std::path::PathBuf,
42    /// Condition variable signaled when copy completes
43    /// Linker tasks wait on this (inside `register_file`) before returning
44    /// Public for testing synchronization behavior
45    pub copy_complete: Arc<compio_sync::Condvar>,
46}
47
48impl std::fmt::Debug for HardlinkInfo {
49    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
50        f.debug_struct("HardlinkInfo")
51            .field("original_path", &self.original_path)
52            .field("inode_number", &self.inode_number)
53            .field("link_count", &self.link_count.load(Ordering::Relaxed))
54            .field("dst_path", &self.dst_path)
55            .finish_non_exhaustive() // Omitting copy_complete condvar
56    }
57}

59/// Filesystem boundary and hardlink tracker
60#[derive(Debug, Default)]
61pub struct FilesystemTracker {
62    /// Tracked hardlinks by inode
63    pub hardlinks: DashMap<InodeInfo, HardlinkInfo>,
64    /// Source filesystem device ID (set during scan to avoid cross-filesystem hardlinks)
65    source_filesystem: std::sync::RwLock<Option<u64>>,
66}

68/// Statistics about hardlink tracking
69#[derive(Debug, Clone, Default)]
70pub struct FilesystemStats {
71    /// Total number of tracked inodes
72    pub total_files: usize,
73    /// Number of inodes with multiple hardlinks
74    pub hardlink_groups: usize,
75    /// Total number of hardlink references
76    pub total_hardlinks: u64,
77}

80impl FilesystemTracker {
81    /// Create a new filesystem tracker
82    #[must_use]
83    pub fn new() -> Self {

85            hardlinks: DashMap::new(),
86            source_filesystem: std::sync::RwLock::new(None),

90    /// Create a tracker with a known source filesystem
91    #[must_use]
92    pub fn with_source_filesystem(dev: u64) -> Self {
93        Self {
94            hardlinks: DashMap::new(),
95            source_filesystem: std::sync::RwLock::new(Some(dev)),
96        }
97    }

99    /// Set the source filesystem device ID
100    ///
101    /// This should be called once at the beginning of a copy operation
102    /// to establish the source filesystem boundary.
103    pub fn set_source_filesystem(&self, dev: u64) {
104        if let Ok(mut source) = self.source_filesystem.write() {
105            *source = Some(dev);
106            debug!("Set source filesystem device ID: {}", dev);
107        }
108    }
109
110    /// Check if a device is on the source filesystem
111    ///
112    /// Returns `true` if the device matches the source filesystem or if no source is set.
113    #[must_use]
114    pub fn is_on_source_filesystem(&self, dev: u64) -> bool {
115        self.source_filesystem
116            .read()
117            .ok()
118            .is_none_or(|source_dev| source_dev.is_none_or(|source| source == dev))
119    }
120
121    /// Check if a device is on the same filesystem as the source
122    ///
123    /// Alias for `is_on_source_filesystem` for compatibility.
124    #[must_use]
125    pub fn is_same_filesystem(&self, dev: u64) -> bool {
126        self.is_on_source_filesystem(dev)
127    }
128
129    /// Check if a path is on the source filesystem
130    pub fn is_path_on_source_filesystem(&self, path: &Path) -> bool {
131        std::fs::metadata(path)
132            .ok()
133            .is_some_and(|m| self.is_on_source_filesystem(m.dev()))
134    }
135
136    /// Check if inode is on source filesystem using a predicate
137    ///
138    /// This is a more flexible version that allows custom device extraction.
139    pub fn check_source_filesystem<F>(&self, dev: u64, predicate: F) -> bool
140    where
141        F: FnOnce(Option<u64>) -> bool,
142    {
143        self.source_filesystem
144            .read()
145            .ok()
146            .is_none_or(|source_dev| match *source_dev {
147                Some(source) if source != dev => false,
148                _ => predicate(*source_dev),
149            })
150    }

152    /// Register a file for hardlink tracking (async - waits if linker)
153    ///
154    /// This atomically determines if this task is the "copier" or a "linker":
155    /// - **Copier**: First to register, returns `true` immediately
156    /// - **Linker**: Already registered, WAITS on condvar until copier signals, returns `false`
157    /// - **Regular file**: `link_count == 1`, returns `false` immediately
158    ///
159    /// The waiting is encapsulated inside this method, so caller doesn't need to manage condvars.
160    ///
161    /// # Arguments
162    /// * `src_path` - Source file path
163    /// * `dst_path` - Destination path (stored in `HardlinkInfo` for linkers to use)
164    /// * `dev` - Device ID  
165    /// * `ino` - Inode number
166    /// * `link_count` - Number of hardlinks (from stat)
167    ///
168    /// # Returns
169    /// * `true` - Caller is copier, should copy file then call `signal_copy_complete()`
170    /// * `false` - Caller is linker (already waited) or regular file, should create hardlink/copy
171    #[allow(clippy::too_many_arguments)]
172    pub async fn register_file(
173        &self,
174        src_path: &Path,
175        dst_path: &Path,
176        dev: u64,
177        ino: u64,
178        link_count: u64,
179    ) -> bool {
180        // Skip files with link count of 1 - they're not hardlinks
181        if link_count == 1 {
182            return false; // Not a hardlink, caller should copy normally
183        }
184
185        let inode_info = InodeInfo { dev, ino };
186
187        // ATOMIC PATTERN: Try to insert, get back whether we won the race
188        match self.hardlinks.entry(inode_info) {
189            dashmap::mapref::entry::Entry::Occupied(entry) => {
190                // This inode is already registered - we're a linker
191                let hardlink_info = entry.get();
192
193                // Increment link count
194                let count = hardlink_info.link_count.fetch_add(1, Ordering::Relaxed);
195
196                // Get condvar before releasing DashMap ref
197                let condvar = Arc::clone(&hardlink_info.copy_complete);
198
199                // Release DashMap ref
200                let _ = hardlink_info;
201                drop(entry);
202
203                debug!(
204                    "Found hardlink #{} for inode ({}, {}): {} (linker - waiting for copy)",
205                    count + 1,
206                    dev,
207                    ino,
208                    src_path.display()
209                );
210
211                // WAIT for copier to signal completion
212                condvar.wait().await;
213
214                debug!(
215                    "Linker woke up for inode ({}, {}): {}",
216                    dev,
217                    ino,
218                    src_path.display()
219                );
220
221                false // We're a linker, waiting is done
222            }
223            dashmap::mapref::entry::Entry::Vacant(entry) => {
224                // We're the first - we're the copier
225                entry.insert(HardlinkInfo {
226                    original_path: src_path.to_path_buf(),
227                    inode_number: ino,
228                    link_count: AtomicU64::new(1),
229                    dst_path: dst_path.to_path_buf(),
230                    copy_complete: Arc::new(compio_sync::Condvar::new()),
231                });
232                debug!(
233                    "Registered new hardlink inode ({}, {}): {} â†’ {} (copier)",
234                    dev,
235                    ino,
236                    src_path.display(),
237                    dst_path.display()
238                );
239                true // We're the copier
240            }
241        }
242    }

244    /// Get all hardlink groups that have multiple links
245    ///
246    /// Returns a vector of (`inode_number`, `link_count`) tuples for inodes with multiple hardlinks.
247    #[must_use]
248    pub fn get_hardlink_groups(&self) -> Vec<(u64, u64)> {
249        self.hardlinks
250            .iter()
251            .filter_map(|entry| {
252                let val = entry.value();
253                let count = val.link_count.load(Ordering::Relaxed);
254                if count > 1 {
255                    Some((val.inode_number, count))
256                } else {
257                    None
258                }
259            })
260            .collect()
261    }

263    /// Signal that an inode's copy is complete
264    ///
265    /// This should be called by the copier task after attempting to copy (success or failure).
266    /// It wakes all waiting linker tasks. Linkers will naturally succeed/fail based on
267    /// whether the destination file exists.
268    ///
269    /// # Arguments
270    /// * `ino` - Inode number that was copied
271    pub fn signal_copy_complete(&self, ino: u64) {
272        // Find the condvar and signal it (move temporary above if-let to avoid significant drop)
273        let found = self
274            .hardlinks
275            .iter()
276            .find(|e| e.value().inode_number == ino);
277        if let Some(entry) = found {
278            let condvar = Arc::clone(&entry.value().copy_complete);
279            drop(entry); // Release DashMap ref before signaling
280
281            condvar.notify_all();
282            debug!("Signaled copy complete for inode {}", ino);
283        }
284    }

286    /// Get the destination path for an inode
287    ///
288    /// Returns the destination path that was set during registration.
289    /// For linkers, this should only be called after `register_file()` returns (waiting is done).
290    #[must_use]
291    pub fn get_dst_path_for_inode(&self, ino: u64) -> Option<PathBuf> {
292        self.hardlinks
293            .iter()
294            .find(|entry| entry.value().inode_number == ino)
295            .map(|entry| entry.value().dst_path.clone())
296    }
297
298    /// Get statistics about the filesystem tracking
299    #[must_use]
300    pub fn get_stats(&self) -> FilesystemStats {
301        let total_files = self.hardlinks.len();
302        let mut hardlink_groups = 0;
303        let mut total_hardlinks = 0;
304
305        for entry in &self.hardlinks {
306            let count = entry.value().link_count.load(Ordering::Relaxed);
307            if count > 1 {
308                hardlink_groups += 1;
309            }
310            total_hardlinks += count;
311        }
312
313        FilesystemStats {
314            total_files,
315            hardlink_groups,
316            total_hardlinks,
317        }
318    }
319
320    /// Check if an inode has been copied
321    #[must_use]
322    pub fn is_inode_copied(&self, dev: u64, ino: u64) -> bool {
323        self.hardlinks.get(&InodeInfo { dev, ino }).is_some()
324    }
325
326    /// Get the original path for an inode if it's been registered
327    #[must_use]
328    pub fn get_original_path_for_inode(&self, dev: u64, ino: u64) -> Option<PathBuf> {
329        self.hardlinks
330            .get(&InodeInfo { dev, ino })
331            .map(|entry| entry.original_path.clone())
332    }

379    /// SYNCHRONIZATION: Test that linker waits for copier using condvar coordination
380    #[compio::test]
381    async fn test_linker_waits_for_copier_signal() {
382        let temp_dir = TempDir::new().expect("Failed to create temp directory");
383        let file1 = temp_dir.path().join("file1.txt");
384        let file2 = temp_dir.path().join("file2.txt");
385        let dst_dir = temp_dir.path().join("dst");
386        std::fs::create_dir(&dst_dir).expect("Failed to create dst dir");
387
388        std::fs::write(&file1, "content").expect("Failed to write file");
389        std::fs::hard_link(&file1, &file2).expect("Failed to create hardlink");
390
391        let meta = std::fs::metadata(&file1).expect("Failed to get metadata");
392        let tracker = Arc::new(FilesystemTracker::new());
393
394        let dst_file1 = dst_dir.join("file1.txt");
395        let dst_file2 = dst_dir.join("file2.txt");
396
397        // Register copier (returns immediately)
398        let is_copier = tracker
399            .register_file(&file1, &dst_file1, 1, meta.ino(), 2)
400            .await;
401        assert!(is_copier, "First registration should be copier");
402
403        // Get inode before spawning task
404        let inode = meta.ino();
405
406        let tracker_clone = Arc::clone(&tracker);
407        let file2_clone = file2.clone();
408        let dst_file2_clone = dst_file2.clone();
409
410        // Use a coordination condvar to ensure linker has time to start waiting
411        let coord_cv = Arc::new(compio_sync::Condvar::new());
412        let coord_cv_clone = Arc::clone(&coord_cv);
413
414        // Spawn a coordinator task that gives the linker time to execute
415        let coord_handle = compio::runtime::spawn(async move {
416            // Yield control to allow linker to start
417            for _ in 0..10 {
418                compio::runtime::spawn(async {}).await.ok();
419            }
420            // Signal that we've given time for linker to enter wait queue
421            coord_cv_clone.notify_all();
422        });
423
424        // Spawn linker task
425        let linker_handle = compio::runtime::spawn(async move {
426            // This will block inside register_file on condvar.wait()
427            let is_linker = tracker_clone
428                .register_file(&file2_clone, &dst_file2_clone, 1, inode, 2)
429                .await;
430            assert!(!is_linker, "Should be linker");
431            "linker woke up"
432        });
433
434        // Wait for coordinator to signal (gives linker time to enter queue)
435        coord_cv.wait().await;
436        coord_handle.await.ok();
437
438        // Note: waiter_count() method removed from compio_sync::Condvar API
439        // The test still verifies that the linker waits and gets notified,
440        // we just can't verify the internal waiter queue state
441
442        // Signal copier completion
443        tracker.signal_copy_complete(inode);
444
445        // Brief sleep to allow notification to propagate
446        std::thread::sleep(std::time::Duration::from_millis(1));
447
448        // Linker should wake up and complete
449        let result = linker_handle.await.expect("Linker task should complete");
450        assert_eq!(result, "linker woke up", "Linker should wake after signal");
451    }
